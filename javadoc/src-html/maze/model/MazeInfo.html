020    /**
021     * Stores information about a Maze including its file name and model.
022     * @author John Smith
023     */
024    public class MazeInfo implements Listener<MazeCell>
025    {
026       private MazeModel mModel;
027       private String mPath;
028       private String mName;
029       private boolean isDirty = false;
030       private boolean isMutable = true;
031       private boolean isExtended = false;
032    
033       public static MazeInfo load(InputStream in, String name)
034       {
035          if (in == null)
036             return null;
037    
038          MazeInfo returned = null;
039          try
040          {
041             MazeInfo mi = new MazeInfo();
042             MazeModel maze = new MazeModel();
043             maze.loadMaze(in, false);
044             mi.mPath = name;
045             mi.isMutable = false;
046             mi.mName = name;
047             mi.mModel = maze;
048             mi.mModel.addDelayedListener(mi);
049             returned = mi;
050          }
051          catch (Exception e)
052          {}
053          return returned;
054       }
055    
056       public static MazeInfo load(File file)
057       {
058          if (file == null || !file.isFile())
059             return null;
060          //DataInputStream dis;
061          MazeInfo returned = null;
062          try
063          {
064             MazeInfo mi = new MazeInfo();
065             MazeModel maze = new MazeModel();
066             mi.mPath = file.getCanonicalPath();
067             String name = maze.loadMaze(mi.mPath);
068             if (name != null)
069             {
070                mi.mName = name;
071                mi.isExtended = true;
072             }
073             else
074                mi.mName = file.getName();
075             mi.mModel = maze;
076    
077             mi.mModel.addDelayedListener(mi);
078             returned = mi;
079          }
080          catch (IOException ex)
081          {}
082    
083          return returned;
084       }
085    
086       public static MazeInfo createEmptyMaze(String name, boolean extended)
087       {
088          MazeInfo mi = new MazeInfo();
089          mi.mName = name;
090          mi.mPath = "";
091          mi.mModel = new MazeModel();
092          mi.isDirty = true;
093          mi.mModel.addDelayedListener(mi);
094          mi.isExtended = extended;
095          return mi;
096       }
097    
098       public boolean store()
099       {
100          try
101          {
102             if (isExtended)
103                mModel.saveMaze(mPath, mName);
104             else
105                mModel.saveMaze(mPath, null);
106             isDirty = false;
107          }
108          catch (IOException e)
109          {
110             System.out.println("Saving Maze Error");
111             return false;
112          }
113    
114          return true;
115       }
116    
117       public String getName()
118       {
119          return mName;
120       }
121    
122       public String getPath()
123       {
124          return mPath;
125       }
126    
127       public MazeModel getModel()
128       {
129          return mModel;
130       }
131    
132       public boolean isDirty()
133       {
134          return isDirty;
135       }
136    
137       public void clearDirty()
138       {
139          if (!isMutable)
140             isDirty = false;
141       }
142    
143       public boolean isMutable()
144       {
145          return isMutable;
146       }
147    
148       public boolean isExtended()
149       {
150          return isExtended;
151       }
152    
153       public void setName(String name)
154       {
155          mName = name;
156          isExtended = true;
157       }
158    
159       public void setPath(String path)
160       {
161          if (!mPath.equalsIgnoreCase(path))
162          {
163             mPath = path;
164             isDirty = true;
165          }
166       }
167    
168       /**
169        * Private constructor. Instances must be retrieved from factory methods.
170        */
171       private MazeInfo()
172       {}
173    
174       @Override
175       public String toString()
176       {
177          return this.mName;
178       }
179    
180       public MazeInfo getMutableClone()
181       {
182          MazeInfo theClone = new MazeInfo();
183          theClone.isDirty = isDirty;
184          theClone.isExtended = isExtended;
185          theClone.isMutable = true;
186          theClone.mModel = (MazeModel) mModel.clone();
187          theClone.mModel.addListener(theClone);
188          theClone.mName = mName;
189          theClone.mPath = mPath;
190          return theClone;
191       }
192    
193       @Override
194       public void eventFired(MazeCell event)
195       {
196          this.isDirty = true;
197       }
198    
199       /**
200        * Save a maze.
201        * @return The saved maze or null. Can return itself or a new instance.
202        */
203       public MazeInfo saveMaze()
204       {
205          final PrimaryFrame frame = Main.getPrimaryFrameInstance();
206          DefaultComboBoxModel cbm = frame.getMazeInfoModel().getMazeInfoComboBoxModel();
207          TreeSet<String> paths = new TreeSet<String>();
208          for (int i = 0; i < cbm.getSize(); i++)
209          {
210             String path = ((MazeInfo) cbm.getElementAt(i)).getPath();
211             if (path.isEmpty() == false)
212                paths.add(path.toLowerCase());
213          }
214          if (this.isDirty())
215          {
216             MazeInfo toSave = this;
217             if (!this.isMutable())
218             {
219                toSave = this.getMutableClone();
220                Box box = new Box(BoxLayout.Y_AXIS);
221                JLabel label = new JLabel("What would you like to call the" + " maze?");
222                JTextField field = new JTextField();
223                box.add(label);
224                box.add(field);
225                String newName = null;
226                while (true)
227                {
228                   int result = JOptionPane.showConfirmDialog(frame,
229                                                              box,
230                                                              "Maze Name",
231                                                              JOptionPane.OK_CANCEL_OPTION,
232                                                              JOptionPane.QUESTION_MESSAGE);
233                   if (result == JOptionPane.OK_OPTION)
234                   {
235                      if (field.getText().length() > 0)
236                      {
237                         newName = field.getText();
238                         break;
239                      }
240                   }
241                   else
242                      break;
243                }
244                if (newName == null)
245                   return null;
246                this.clearDirty();
247                toSave.setName(newName);
248                toSave.setPath("");
249                frame.getMazeInfoModel().addMaze(toSave);
250                toSave.saveMaze();
251                return toSave;
252             }
253             else if (this.getPath() == null || this.getPath().isEmpty())
254             {
255                JFileChooser chooser = new JFileChooser();
256                while (true)
257                {
258                   if (chooser.showSaveDialog(frame) == JFileChooser.APPROVE_OPTION)
259                   {
260                      File file = chooser.getSelectedFile();
261                      try
262                      {
263                         if (paths.contains(file.getCanonicalPath().toLowerCase()))
264                         {
265                            JOptionPane.showMessageDialog(frame,
266                                                          "That file is being used by another maze",
267                                                          "Error",
268                                                          JOptionPane.ERROR_MESSAGE);
269                         }
270                         else
271                         {
272                            this.setPath(file.getCanonicalPath());
273                            break;
274                         }
275                      }
276                      catch (IOException ex)
277                      {
278                         JOptionPane.showMessageDialog(frame,
279                                                       "Invalid Save File",
280                                                       "Error",
281                                                       JOptionPane.ERROR_MESSAGE);
282                      }
283                   }
284                   else
285                   { // Save dialog not approved.
286                      break;
287                   }
288                } // End infinite loop.
289             }
290             this.store();
291          } // if (mi.isDirty())
292          return this;
293       }
294    }
