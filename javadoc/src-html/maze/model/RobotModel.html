001    package maze.model;
002    
003    /**
004     * This is a wrapper class for {@link RobotModelMaster}.<br />
005     * This class exists to provide limited access to the data models for the AI
006     * algorithms.
007     * @author Luke Last
008     */
009    public class RobotModel
010    {
011       private final RobotModelMaster parent;
012    
013       public RobotModel(RobotModelMaster parent)
014       {
015          this.parent = parent;
016       }
017    
018       public java.awt.Dimension getMazeSize()
019       {
020          return this.parent.getMazeSize();
021       }
022    
023       public boolean isWallFront()
024       {
025          return this.parent.isWall(this.getDirection());
026       }
027    
028       public boolean isWallBack()
029       {
030          return this.parent.isWall(this.getDirection().getOpposite());
031       }
032    
033       public boolean isWallLeft()
034       {
035          return this.parent.isWall(this.getDirection().getLeft());
036       }
037    
038       public boolean isWallRight()
039       {
040          return this.parent.isWall(this.getDirection().getRight());
041       }
042    
043       public MazeCell getCurrentLocation()
044       {
045          return this.parent.getCurrentLocation();
046       }
047    
048       public Direction getDirection()
049       {
050          return this.parent.getDirection();
051       }
052    }
