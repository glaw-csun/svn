001    package maze.util;
002    
003    import java.awt.EventQueue;
004    
005    /**
006     * @author Luke Last
007     * @param <T> The type of the event objects to be used by the Listeners.
008     */
009    public class ListenerSubject<T>
010    {
011       /**
012        * Zero, one, or many listeners. Can be null. Multiple listeners form a
013        * linked list.
014        */
015       private Listener<T> listener;
016    
017       /**
018        * Stores listeners that should be notified in a delayed manner.
019        */
020       private Listener<T> delayedListener;
021    
022       /**
023        * Stores whether or not a delayed notify Runnable has been created on the
024        * EventQueue but not yet executed.
025        */
026       private volatile boolean pendingNotify = false;
027    
028       /**
029        * Notifies all registered listeners.
030        * @param event The event object that should be passed to each registered
031        *           Listener.
032        */
033       public void notifyListeners(final T event)
034       {
035          final Listener<T> copy = this.listener;
036          if (copy != null)
037          {
038             copy.eventFired(event);
039          }
040          if (this.delayedListener != null && this.pendingNotify == false)
041          {
042             this.pendingNotify = true;
043             EventQueue.invokeLater(new Runnable()
044             {
045                @Override
046                public void run()
047                {
048                   pendingNotify = false;
049                   final Listener<T> delayedCopy = delayedListener;
050                   if (delayedCopy != null)
051                   {
052                      delayedCopy.eventFired(event);
053                   }
054                }
055             });
056          }
057       }
058    
059       /**
060        * Registers a Listener with this subject. The given Listener is not checked
061        * to see if it already is registered.
062        * @param listener The Listener you would like to add.
063        */
064       public synchronized void addListener(final Listener<T> listener)
065       {
066          if (listener != null)
067          {
068             if (this.listener == null)
069             {
070                this.listener = listener;
071             }
072             else
073             {
074                this.listener = new ListenerMulticaster<T>(this.listener, listener);
075             }
076          }
077       }
078    
079       /**
080        * @param listener
081        */
082       public synchronized void addDelayedListener(final Listener<T> listener)
083       {
084          if (listener != null)
085          {
086             if (this.delayedListener == null)
087             {
088                this.delayedListener = listener;
089             }
090             else
091             {
092                this.delayedListener = new ListenerMulticaster<T>(this.delayedListener, listener);
093             }
094          }
095       }
096    
097       /**
098        * Unregisters a Listener with this subject.
099        * @param listener The Listener you want to remove.
100        */
101       public synchronized void removeListener(final Listener<T> listener)
102       {
103          if (this.listener != null && listener != null)
104          {
105             if (this.listener == listener)
106             {
107                this.listener = null;
108             }
109             else if (this.listener instanceof ListenerMulticaster)
110             {
111                final ListenerMulticaster<T> multicaster = (ListenerMulticaster<T>) this.listener;
112                this.listener = multicaster.remove(listener);
113             }
114          }
115       }
116    
117       public synchronized void removeAllListeners()
118       {
119          this.listener = null;
120       }
121    
122       /**
123        * A utility class used to join 2 Listeners into one Listener interface. It
124        * is used to create a readonly linked list of Listeners.<br />
125        * Each instance is immutable.
126        * @author Luke Last
127        * @param <T>
128        */
129       private static class ListenerMulticaster<T> implements Listener<T>
130       {
131          private final Listener<T> a;
132          private final Listener<T> b;
133    
134          /**
135           * @param a First child branch.
136           * @param b Second child branch.
137           */
138          private ListenerMulticaster(Listener<T> a, Listener<T> b)
139          {
140             this.a = a;
141             this.b = b;
142          }
143    
144          /**
145           * Removes a Listener from this linked list. Because we are immutable new
146           * instances must be created when removing an object.
147           * @param toRemove The Listener to be removed.
148           * @return The new Listener resulting from the removal. Can return itself
149           *         if nothing changed.
150           */
151          private Listener<T> remove(Listener<T> toRemove)
152          {
153             if (this.a == toRemove)
154             {
155                return this.b;
156             }
157             if (this.b == toRemove)
158             {
159                return this.a;
160             }
161             final Listener<T> a2;
162             final Listener<T> b2;
163             if (this.a instanceof ListenerMulticaster)
164             {
165                a2 = ((ListenerMulticaster<T>) this.a).remove(toRemove);
166                if (this.b instanceof ListenerMulticaster)
167                {
168                   b2 = ((ListenerMulticaster<T>) this.b).remove(toRemove);
169                }
170                else
171                {
172                   b2 = this.b;
173                }
174             }
175             else
176             {
177                a2 = this.a;
178                if (this.b instanceof ListenerMulticaster)
179                {
180                   b2 = ((ListenerMulticaster<T>) this.b).remove(toRemove);
181                }
182                else
183                {
184                   return this;
185                }
186             }
187             return new ListenerMulticaster<T>(a2, b2);
188          }
189    
190          /**
191           * Passes the event on to children.
192           */
193          @Override
194          public void eventFired(T event)
195          {
196             if (this.a != null)
197             {
198                this.a.eventFired(event);
199             }
200             if (this.b != null)
201             {
202                this.b.eventFired(event);
203             }
204          }
205       }
206    }
