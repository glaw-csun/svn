001    package maze.util;
002    
003    /**
004     * This Listener interface allows the creation of Listener classes that can
005     * register themselves with a ListenerSubject.
006     * @author Luke Last
007     * @param <T> The type of the event object that is passed in when the event is
008     *           fired. This type should match the defined generic type of the
009     *           ListenerSubject you want to register with.
010     */
011    public interface Listener<T>
012    {
013       /**
014        * This is called when the ListenerSubject wants to notify the registered
015        * Listeners.
016        * @param event The event object.
017        */
018       public void eventFired(T event);
019    }
