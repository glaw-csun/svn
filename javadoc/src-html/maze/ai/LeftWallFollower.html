001    package maze.ai;
002    
003    import java.util.ArrayList;
004    import java.util.List;
005    
006    /**
007     * This is the interface for the robot AI that is used by the GUI.
008     */
009    public class LeftWallFollower extends RobotBase
010    {
011       private List<RobotStep> moveQueue = new ArrayList<RobotStep>();
012    
013       @Override
014       public String toString()
015       {
016          return "Left Wall Follower";
017       }
018    
019       /**
020        * Called to before any steps are requested of the robot. This can be
021        * overridden to do any initialization of the AI.
022        */
023       @Override
024       public void initialize()
025       {
026          super.initialize();
027          moveQueue.clear();
028       }
029    
030       /**
031        * This is called by the controller to get the next move that the robot AI
032        * will take.
033        */
034       @Override
035       public RobotStep nextStep()
036       {
037          RobotStep next;
038          if (moveQueue.isEmpty() == true)
039          {
040             if (robotLocation.isWallLeft() == false)
041             {
042                next = RobotStep.RotateLeft;
043                moveQueue.add(RobotStep.MoveForward);
044             }
045             else if (robotLocation.isWallFront() == false)
046             {
047                next = RobotStep.MoveForward;
048             }
049             else
050             {
051                next = RobotStep.RotateRight;
052             }
053          }
054          else
055          {
056             next = moveQueue.get(0);
057             moveQueue.remove(0);
058          }
059          return next;
060       }
061    }
