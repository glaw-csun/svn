001    package maze.ai;
002    
003    import maze.model.Direction;
004    import maze.model.MazeCell;
005    import maze.model.MazeModel;
006    import maze.model.RobotModel;
007    import maze.model.RobotModelMaster;
008    
009    /**
010     * Controls the AI and models to move the robot through the maze.
011     * @author Luke Last
012     */
013    public class RobotController
014    {
015       private static final int MAX_STEP_COUNT = 1500;
016       private final MazeModel mazeModel;
017       private final RobotModelMaster robotModelMaster;
018       private final RobotModel robotModelClient;
019       private final RobotBase ai;
020       private int robotStepCount = 0;
021    
022       /**
023        * Constructor.
024        * @param model The maze model to use.
025        * @param robotAI The robot AI algorithm to use.
026        */
027       public RobotController(MazeModel model, RobotBase robotAI)
028       {
029          this.mazeModel = model;
030          this.ai = robotAI;
031    
032          final MazeCell start = new MazeCell(1, this.mazeModel.getSize().height);
033    
034          this.robotModelMaster = new RobotModelMaster(this.mazeModel, start, Direction.North);
035    
036          this.robotModelClient = new RobotModel(this.robotModelMaster);
037          this.initialize();
038       }
039    
040       /**
041        * Initializes/resets all values of this controller allowing a new simulation
042        * to be run from the beginning.
043        */
044       public void initialize()
045       {
046          this.robotModelMaster.setCurrentLocation(new MazeCell(1, this.mazeModel.getSize().height));
047          this.robotModelMaster.setDirection(Direction.North);
048          this.ai.setRobotLocation(this.robotModelClient);
049          this.ai.initialize();
050       }
051    
052       /**
053        * Get the next step the robot has taken.
054        */
055       public RobotStep nextStep()
056       {
057          final RobotStep nextStep = this.ai.nextStep();
058          //System.out.println(nextStep);
059    
060          this.robotModelMaster.takeNextStep(nextStep);
061    
062          this.robotStepCount++;
063          return nextStep;
064       }
065    
066       /**
067        * Is the robot done moving.
068        */
069       public boolean isRobotDone()
070       {
071          return this.robotStepCount > MAX_STEP_COUNT;
072       }
073    
074       /**
075        * Get the total number of steps the robot has taken so far. Turns count as a
076        * step.
077        */
078       public int getStepCount()
079       {
080          return this.robotStepCount;
081       }
082    
083       /**
084        * Get the robot model being used by this controller.
085        */
086       public RobotModelMaster getRobotModelMaster()
087       {
088          return this.robotModelMaster;
089       }
090    
091       public Direction[][] getUnderstandingDir()
092       {
093          return ai.getUnderstandingDir();
094       }
095    
096       public int[][] getUnderstandingInt()
097       {
098          return ai.getUnderstandingInt();
099       }
100    
101    }
