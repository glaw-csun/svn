037       /**
038        * Initializes the algorithm for use.  This should be called before any run
039        * is started.
040        */
041       public void initialize()
042       {
043          super.initialize();
044          moveQueue.clear();
045          maze.setSize(robotLocation.getMazeSize());
046          maze.clearMaze();
047          Dimension size = maze.getSize();
048          if (distance == null)
049          {
050             distance = new int[size.width][size.height];
051             explored = new boolean[size.width][size.height];
052          }
053          for (int i = 0; i < size.getWidth(); i++)
054          {
055             for (int j = 0; j < size.getHeight(); j++)
056             {
057                distance[i][j] = USELESS;
058                explored[i][j] = false;
059             }
060          }
061          goal = TO_CENTER;
062          speedRunCapable = false;
063       }

065       @Override
066       /**
067        * Returns the next step that the robot should take.  This is called by the
068        * controller.
069        */
070       public RobotStep nextStep()
071       {
072          RobotStep next;
073          Direction nextDirection;
074          Direction currentDirection = robotLocation.getDirection();
075          if (moveQueue.isEmpty())
076          {
077             if (getExplored() == false)
078             {
079                checkWalls();
080                setExplored();
081             }
082             if (atGoal() == true)
083             {
084                if ( (goal == TO_CENTER) && (speedRunCapable == false))
085                {
086                   speedRunCapable = true;
087                   blockOutCenter();
088                }
089                goal = !goal;
090                floodfill();
091             }
092             nextDirection = getBestDirection();
093             turbo = getNeighborExplored(nextDirection);
094             if (nextDirection == currentDirection)
095             {
096                next = RobotStep.MoveForward;
097             }
098             else if (nextDirection == currentDirection.getLeft())
099             {
100                next = RobotStep.RotateLeft;
101                moveQueue.add(RobotStep.MoveForward);
102             }
103             else if (nextDirection == currentDirection.getRight())
104             {
105                next = RobotStep.RotateRight;
106                moveQueue.add(RobotStep.MoveForward);
107             }
108             else
109             // if(nextDirection == currentDirection.getOpposite())
110             {
111                next = RobotStep.MoveBackward;
112             }
113          }
114          else
115          {
116             next = moveQueue.get(0);
117             moveQueue.remove(0);
118          }
119          return next;
120       }

122       /**
123        * Returns the explored flag of the current location's neighbor in the 
124        * given direction.
125        */
126       private boolean getNeighborExplored(Direction direction)
127       {
128          MazeCell neighbor;
129          MazeCell here = robotLocation.getCurrentLocation();
130          Dimension size = maze.getSize();
131          if ( (direction == Direction.North) && (here.getY() != 1))
132          {
133             neighbor = new MazeCell(here.getX(), here.getY() - 1);
134          }
135          else if ( (direction == Direction.South) 
136                      && (here.getY() != size.getHeight()))
137          {
138             neighbor = new MazeCell(here.getX(), here.getY() + 1);
139          }
140          else if ( (direction == Direction.East) 
141                      && (here.getX() != size.getWidth()))
142          {
143             neighbor = new MazeCell(here.getX() + 1, here.getY());
144          }
145          else if ( (direction == Direction.West) && (here.getX() != 1))
146          {
147             neighbor = new MazeCell(here.getX() - 1, here.getY());
148          }
149          else
150          {
151             return false;
152          }
153    
154          return getExplored(neighbor);
155       }

206       /**

215       /**
216        * Returns the best direction to go, according to Floodfill, from the 
217        * current MazeCell.  This algorithm biases in the following order:
218        * Straight, North, East, West, South
219        */
220       private Direction getBestDirection()
221       {
222          MazeCell here = robotLocation.getCurrentLocation();
223          int bestDistance = getDistance(here);
224          Direction bestDirection = null;
225    
226          if ( (bestDistance > 
227          getNeighborDistance(here, robotLocation.getDirection())) &&
228              (robotLocation.isWallFront() == false))
229          {
230             bestDirection = robotLocation.getDirection();
231             bestDistance = getNeighborDistance(here, bestDirection);
232          }
233    
234          if ( (bestDistance > getNeighborDistance(here, Direction.North)) &&
235              (maze.getWall(here, Direction.North).isSet() == false))
236          {
237             bestDirection = Direction.North;
238             bestDistance = getNeighborDistance(here, bestDirection);
239          }
240          if ( (bestDistance > getNeighborDistance(here, Direction.East)) &&
241              (maze.getWall(here, Direction.East).isSet() == false))
242          {
243             bestDirection = Direction.East;
244             bestDistance = getNeighborDistance(here, bestDirection);
245          }
246          if ( (bestDistance > getNeighborDistance(here, Direction.West)) &&
247              (maze.getWall(here, Direction.West).isSet() == false))
248          {
249             bestDirection = Direction.West;
250             bestDistance = getNeighborDistance(here, bestDirection);
251          }
252          if ( (bestDistance > getNeighborDistance(here, Direction.South)) &&
253              (maze.getWall(here, Direction.South).isSet() == false))
254          {
255             bestDirection = Direction.South;
256             bestDistance = getNeighborDistance(here, bestDirection);
257          }
258    
259          if (bestDirection == null)
260          {
261             floodfill();
262             return getBestDirection();
263          }
264          else
265          {
266             return bestDirection;
267          }
268       }

270       /**
271        * Returns the distance of the MazeCell adjacent to the passed MazeCell
272        * and is adjacent in the specified direction.
273        */
274       private int getNeighborDistance(MazeCell here, Direction direction)
275       {
276          MazeCell neighbor;
277          Dimension size = maze.getSize();
278          if ( (direction == Direction.North) && (here.getY() != 1))
279          {
280             neighbor = new MazeCell(here.getX(), here.getY() - 1);
281          }
282          else if ( (direction == Direction.South) 
283                      && (here.getY() != size.getHeight()))
284          {
285             neighbor = new MazeCell(here.getX(), here.getY() + 1);
286          }
287          else if ( (direction == Direction.East) 
288                      && (here.getX() != size.getWidth()))
289          {
290             neighbor = new MazeCell(here.getX() + 1, here.getY());
291          }
292          else if ( (direction == Direction.West) && (here.getX() != 1))
293          {
294             neighbor = new MazeCell(here.getX() - 1, here.getY());
295          }
296          else
297          {
298             return USELESS;
299          }
300    
301          return getDistance(neighbor);
302       }

320       private void floodfill()
321       {
322          Dimension size = maze.getSize();
323          Vector<MazeCell> queue = new Vector<MazeCell>();
324          MazeCell cell;
325          int currentDistance;
326          boolean speedy;
327    
328          for (int i = 1; i <= size.width; i++)
329          {
330             for (int j = 1; j <= size.height; j++)
331             {
332                setDistance(new MazeCell(i, j), USELESS);
333             }
334          }
335    
336          if (goal == TO_START)
337          {
338             cell = new MazeCell(1, size.height);
339             setDistance(cell, 0);
340             queue.add(cell);
341             speedy = false;
342          }
343          else
344          {
345             int targetX = size.width / 2;
346             int targetY = size.height / 2;
347             cell = new MazeCell(targetX, targetY);
348             setDistance(cell, 0);
349             queue.add(cell);
350             cell = new MazeCell(targetX + 1, targetY);
351             setDistance(cell, 0);
352             queue.add(cell);
353             cell = new MazeCell(targetX, targetY + 1);
354             setDistance(cell, 0);
355             queue.add(cell);
356             cell = new MazeCell(targetX + 1, targetY + 1);
357             setDistance(cell, 0);
358             queue.add(cell);
359             if ( (speedRun == true) && (speedRunCapable == true))
360             {
361                speedy = true;
362             }
363             else
364             {
365                speedy = false;
366             }
367          }
368    
369          while (queue.isEmpty() == false)
370          {
371             cell = queue.get(0);
372             queue.remove(0);
373             currentDistance = getDistance(cell);
374    
375             //Check to see if accessible
376             if (maze.getWall(cell, Direction.North).isSet() == false)
377             { //Check to see if it should be added to queue
378                if ( ( (currentDistance + 1) < 
379                            getNeighborDistance(cell, Direction.North)) &&
380                    ( (speedy == false) || 
381                            (getNeighborExplored(cell, Direction.North) == true)))
382                {
383                   queue.add(cell.plusY(-1));
384                   setDistance(cell.plusY(-1), currentDistance + 1);
385                }
386             }
387    
388             //Check to see if accessible
389             if (maze.getWall(cell, Direction.South).isSet() == false)
390             { //Check to see if it should be added to queue
391                if ( ( (currentDistance + 1) < 
392                            getNeighborDistance(cell, Direction.South)) &&
393                    ( (speedy == false) || 
394                                    (getNeighborExplored(cell, Direction.South))))
395                {
396                   queue.add(cell.plusY(1));
397                   setDistance(cell.plusY(1), currentDistance + 1);
398                }
399             }
400    
401             //Check to see if accessible
402             if (maze.getWall(cell, Direction.West).isSet() == false)
403             { //Check to see if it should be added to queue
404                if ( ( (currentDistance + 1) < 
405                            getNeighborDistance(cell, Direction.West)) &&
406                    ( (speedy == false) || 
407                                    (getNeighborExplored(cell, Direction.West))))
408                {
409                   queue.add(cell.plusX(-1));
410                   setDistance(cell.plusX(-1), currentDistance + 1);
411                }
412             }
413    
414             //Check to see if accessible
415             if (maze.getWall(cell, Direction.East).isSet() == false)
416             { //Check to see if it should be added to queue
417                if ( ( (currentDistance + 1) < 
418                            getNeighborDistance(cell, Direction.East)) &&
419                    ( (speedy == false) || 
420                                    (getNeighborExplored(cell, Direction.East))))
421                {
422                   queue.add(cell.plusX(1));
423                   setDistance(cell.plusX(1), currentDistance + 1);
424                }
425             }
426    
427          }
428    
429          MazeCell here = robotLocation.getCurrentLocation();
430          if (getDistance(here) == USELESS)
431          {
432             System.out.println("Purging Knowledge");
433             maze.clearMaze();
434             speedRunCapable = false;
435             for (int i = 0; i < size.width; i++)
436             {
437                for (int j = 0; j < size.height; j++)
438                {
439                   explored[i][j] = false;
440                }
441             }
442             explored[here.getX() - 1][here.getY() - 1] = true;
443             checkWalls();
444             floodfill();
445          }
446       }

448       /**
449        * This helper function should only be called when first entering the 
450        * center goal.  Sets the maze knowledge of the algorithm to think that
451        * there are no other ways out of the center because according the American
452        * rules there is only one entrance/exit associated with the goal. 
453        */
454       private void blockOutCenter()
455       {
456          Dimension size = maze.getSize();
457          MazeCell cell1 = new MazeCell(size.width / 2, size.height / 2);
458          MazeCell cell2 = new MazeCell(size.width / 2 + 1, size.height / 2);
459          MazeCell cell3 = new MazeCell(size.width / 2, size.height / 2 + 1);
460          MazeCell cell4 = new MazeCell(size.width / 2 + 1, size.height / 2 + 1);
461          MazeCell current = robotLocation.getCurrentLocation();
462    
463          if (cell1.equals(current) == false)
464          {
465             maze.setWall(cell1.getX(), cell1.getY(), Direction.North.getIndex());
466             maze.setWall(cell1.getX(), cell1.getY(), Direction.West.getIndex());
467          }
468          if (cell2.equals(current) == false)
469          {
470             maze.setWall(cell2.getX(), cell2.getY(), Direction.North.getIndex());
471             maze.setWall(cell2.getX(), cell2.getY(), Direction.East.getIndex());
472          }
473          if (cell3.equals(current) == false)
474          {
475             maze.setWall(cell3.getX(), cell3.getY(), Direction.South.getIndex());
476             maze.setWall(cell3.getX(), cell3.getY(), Direction.West.getIndex());
477          }
478          if (cell4.equals(current) == false)
479          {
480             maze.setWall(cell4.getX(), cell4.getY(), Direction.South.getIndex());
481             maze.setWall(cell4.getX(), cell4.getY(), Direction.East.getIndex());
482          }
483       }

485       /**
486        * Returns true if the current location of the robot is its current destination.
487        */
488       private boolean atGoal()
489       {
490          MazeCell cell = robotLocation.getCurrentLocation();
491          Dimension size = maze.getSize();
492          if ( (goal == TO_START) && (cell.getX() == 1) 
493                      && (cell.getY() == size.height))
494          {
495             return true;
496          }
497          if ( (goal == TO_CENTER) &&
498              ( (cell.getY() == size.height / 2) 
499                              || (cell.getY() == (size.height / 2 + 1))) &&
500              ( (cell.getX() == size.width / 2) 
501                              || (cell.getX() == (size.width / 2 + 1))))
502          {
503             return true;
504          }
505          return false;
506       }

508       /**
509        * Updates the algorithm's memory of the maze based upon the walls in the
510        * current cell.
511        */
512       private void checkWalls()
513       {
514          MazeCell cell = robotLocation.getCurrentLocation();
515          Direction direction = robotLocation.getDirection();
516          if (robotLocation.isWallFront())
517          {
518             maze.setWall(cell.getX(), cell.getY(), direction.getIndex());
519          }
520          if (robotLocation.isWallLeft())
521          {
522             maze.setWall(cell.getX(), cell.getY(), 
523                             direction.getLeft().getIndex());
524          }
525          if (robotLocation.isWallRight())
526          {
527             maze.setWall(cell.getX(), cell.getY(),
528                             direction.getRight().getIndex());
529          }
530          if (robotLocation.isWallBack())
531          {
532             maze.setWall(cell.getX(), cell.getY(),
533                             direction.getOpposite().getIndex());
534          }
535       }
