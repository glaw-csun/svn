001    package maze;

003    import java.awt.image.BufferedImage;
004    
005    import javax.swing.ImageIcon;
006    import javax.swing.JFrame;
007    import javax.swing.JOptionPane;
008    import javax.swing.SwingUtilities;
009    import javax.swing.UIManager;
010    
011    import maze.gui.PrimaryFrame;
012    
013    /**
014     * The main starting point of the application.
015     */
016    public final class Main
017    {
018       /**
019        * Holds a reference to the instance of the primary frame.
020        */
021       private static PrimaryFrame primaryFrame;
022    
023       /**
024        * A global way to get an instance of the applications primary frame.
025        */
026       public static PrimaryFrame getPrimaryFrameInstance()
027       {
028          return primaryFrame;
029       }
030    
031       /**
032        * Classic main entry point function.
033        */
034       public static void main(String[] args)
035       {
036          //Set the GUI to be constructed on the Swing Event Dispatch Thread.
037          SwingUtilities.invokeLater(new Runnable()
038          {
039             @Override
040             public void run()
041             {
042                try
043                {
044                   //System.out.println(Arrays.toString(UIManager.getInstalledLookAndFeels()));
045                   //UIManager.setLookAndFeel(UIManager.getSystemLookAndFeelClassName());
046                   //UIManager.setLookAndFeel(UIManager.getCrossPlatformLookAndFeelClassName());
047                   //UIManager.setLookAndFeel("com.sun.java.swing.plaf.windows.WindowsClassicLookAndFeel");
048                   //UIManager.setLookAndFeel("javax.swing.plaf.metal.MetalLookAndFeel");
049                   UIManager.setLookAndFeel("com.sun.java.swing.plaf.nimbus.NimbusLookAndFeel");
050                }
051                catch (Exception e)
052                {
053                   e.printStackTrace();
054                }
055                primaryFrame = new PrimaryFrame();
056                primaryFrame.init();
057                primaryFrame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
058                primaryFrame.setVisible(true);
059             }
060          });
061       }
062    
063       /**
064        * Utility method used to get image resources for button images while
065        * handling failures gracefully.
066        * @param imagePath The path to the image relative to the location of this
067        *           class. So starting in the 'maze' package.
068        */
069       public static ImageIcon getImageResource(String imagePath)
070       {
071          try
072          {
073             return new ImageIcon(Main.class.getResource(imagePath));
074          }
075          catch (Exception e)
076          {
077             e.printStackTrace();
078             JOptionPane.showMessageDialog(null,
079                                           "Cannot Load Image: " + imagePath + "\nException: " + e,
080                                           "Error Loading Resource",
081                                           JOptionPane.ERROR_MESSAGE);
082             return new ImageIcon(new BufferedImage(10,10,BufferedImage.TYPE_INT_RGB));
083          }
084       }
085    }
