006    package maze.gui.mazeeditor;

009    /**
010     *
011     * @author Johnathan Smith
012     */
013    public class StaggeredTemplate extends ConjoinedMazeTemplate
014    {
015       private static final int MIN_SIZE = 0;
016       protected int mSize = MIN_SIZE;
017       protected int rotation = 0;
018    
019       public StaggeredTemplate()
020       {
021          this.mIcon = maze.Main.getImageResource("gui/mazeeditor/images/Staggered.png");
022          this.mDesc = "Staggered";
023          updateTemplate();
024       }
025    
026       @Override
027       public void nextOrientation()
028       {
029          rotation = ((rotation+1)%4);
030          updateTemplate();
031       }
032    
033       @Override
034       public void grow()
035       {
036          mSize++;
037          updateTemplate();
038       }
039    
040       @Override
041       public void shrink()
042       {
043          if (mSize > MIN_SIZE)
044          {
045             mSize--;
046             updateTemplate();
047          }
048       }
049    
050       @Override
051       public void reset()
052       {
053          rotation = 0;
054          mSize = MIN_SIZE;
055          updateTemplate();
056       }
057    
058       protected void updateTemplate()
059       {
060          int positive = mSize/2 + mSize%2;
061          int negative = mSize/2;
062          mCenter = new TemplatePeg();
063          TemplatePeg lastPos, lastNeg;
064          TemplatePeg newPeg;
065          TemplateWall newWall;
066          lastPos = lastNeg = mCenter;
067    
068          lastPos = createCHalf(mCenter, true);
069          lastNeg = createCHalf(mCenter, false);
070    
071          while(positive > 0)
072          {
073             newPeg = new TemplatePeg();
074             newWall = new TemplateWall();
075             if (rotation < 2)
076             {
077                newWall.mLeftBottom = lastPos;
078                newWall.mRightTop = newPeg;
079             }
080             else
081             {
082                newWall.mRightTop = lastPos;
083                newWall.mLeftBottom = newPeg;
084             }
085    
086             switch(rotation)
087             {
088                case 0:
089                   lastPos.top = newPeg.bottom = newWall;
090                   break;
091                case 1:
092                   lastPos.right = newPeg.left = newWall;
093                   break;
094                case 2:
095                   lastPos.bottom = newPeg.top = newWall;
096                   break;
097                case 3:
098                   lastPos.left = newPeg.right = newWall;
099             }
100             lastPos = createCHalf(newPeg, true);
101             positive--;
102          }
103    
104          while (negative > 0)
105          {
106             newPeg = new TemplatePeg();
107             newWall = new TemplateWall();
108             if (rotation < 2)
109             {
110                newWall.mRightTop = lastNeg;
111                newWall.mLeftBottom = newPeg;
112             }
113             else
114             {
115                newWall.mLeftBottom = lastNeg;
116                newWall.mRightTop = newPeg;
117             }
118    
119             switch(rotation)
120             {
121                case 0:
122                   lastNeg.bottom = newPeg.top = newWall;
123                   break;
124                case 1:
125                   lastNeg.left = newPeg.right = newWall;
126                   break;
127                case 2:
128                   lastNeg.top = newPeg.bottom = newWall;
129                   break;
130                case 3:
131                   lastNeg.right = newPeg.left = newWall;
132                   break;
133             }
134             lastNeg = createCHalf(newPeg, false);
135             negative--;
136          }
137       }
138    
139       protected TemplatePeg createCHalf(TemplatePeg center, boolean positive)
140       {
141          TemplatePeg pegs[] = new TemplatePeg[2];
142          TemplateWall walls[] = new TemplateWall[2];
143    
144          for(int i = 0; i < 2; i++)
145          {
146             pegs[i] = new TemplatePeg();
147             walls[i] = new TemplateWall();
148          }
149    
150          if (rotation == 0 || rotation == 3)
151          {
152             walls[1].mLeftBottom = pegs[0];
153             walls[1].mRightTop = pegs[1];
154          }
155          else
156          {
157             walls[1].mLeftBottom = pegs[1];
158             walls[1].mRightTop = pegs[0];
159          }
160    
161          if (positive &&(rotation == 0 || rotation == 1) ||
162              !positive && (rotation == 2 || rotation == 3))
163          {
164             walls[0].mLeftBottom = center;
165             walls[0].mRightTop = pegs[0];
166          }
167          else
168          {
169             walls[0].mLeftBottom = pegs[0];
170             walls[0].mRightTop = center;
171          }
172    
173          switch(rotation)
174          {
175             case 0:
176                if (positive)
177                {
178                   center.top = walls[0];
179                   pegs[0].bottom = walls[0];
180                }
181                else
182                {
183                   center.bottom = walls[0];
184                   pegs[0].top = walls[0];
185                }
186                pegs[0].right = pegs[1].left = walls[1];
187                break;
188             case 1:
189                if (positive)
190                {
191                   center.right = walls[0];
192                   pegs[0].left = walls[0];
193                }
194                else
195                {
196                   center.left = walls[0];
197                   pegs[0].right = walls[0];
198                }
199                pegs[0].bottom = pegs[1].top = walls[1];
200                break;
201             case 2:
202                if (positive)
203                {
204                   center.bottom = walls[0];
205                   pegs[0].top = walls[0];
206                }
207                else
208                {
209                   center.top = walls[0];
210                   pegs[0].bottom = walls[0];
211                }
212                pegs[0].left = pegs[1].right = walls[1];
213                break;
214             case 3:
215                if (positive)
216                {
217                   center.left = walls[0];
218                   pegs[0].right = walls[0];
219                }
220                else
221                {
222                   center.right = walls[0];
223                   pegs[0].left = walls[0];
224                }
225                pegs[0].top = pegs[1].bottom = walls[1];
226                break;
227          }
228          return pegs[0];
229       }
