001    package maze.gui.mazeeditor;
002    
003    import java.awt.Color;
004    import java.awt.Dimension;
005    import java.awt.Graphics;
006    import java.awt.Graphics2D;
007    import java.awt.Point;
008    import java.awt.event.ComponentEvent;
009    import java.util.TreeSet;
010    import java.util.Vector;
011    import maze.gui.CellSize;
012    import maze.model.Direction;
013    import maze.model.MazeCell;
014    import maze.model.MazeModel.MazeWall;
015    
016    /**
017     * This class provides a MazeView on which maze templates can be applied to the
018     * underlying model.  This view is repsonsible for drawing the current template
019     * that will be applied.
020     * @author Johnathan Smith
021     */
022    public class EditableMazeView extends maze.gui.MazeView
023    {
024       private static final int WALL_SIZE_DIVIDER = 3;
025       private MazeTemplate mCurrentTemplate = null;
026    
027       /**
028        * Returns a CellSize object that hass the current size and dimensions of
029        * the maze cells and walls the whole maze based on the size of this
030        * MazeView panel.
031        * @return a CellSize object with the cell, walls and maze sizes
032        */
033       public CellSize getCellSize()
034       {
035          return new CellSize(csm.getCellWidth()-csm.getWallWidth(),
036                              csm.getCellHeight()-csm.getWallHeight(),
037                              csm.getWallWidth(), csm.getWallHeight(),
038                              csm.getCellWidth()*model.getSize().width,
039                              csm.getCellHeight()*model.getSize().height);
040       }
041    
042       /**
043        * Resizes the wall width and height of the maze based on the size of this
044        * MazeView panel when the panel is resized.
045        * @param e the ComponentEvent passed when this component is resized
046        */
047       @Override
048       public void componentResized(ComponentEvent e)
049       {
050          super.componentResized(e);
051          //We override the standard wall size here so it is easier to click on.
052          final int wallSize = Math.min(csm.getCellWidth(),
053                                  csm.getCellHeight()) / WALL_SIZE_DIVIDER;
054          this.csm.setWallWidth(wallSize);
055          this.csm.setWallHeight(wallSize);
056       }
057    
058       /**
059        * Paints the maze with a call to MazeView.paintComponent and paints the
060        * currently selected maze template if one is selected.
061        * @param arg the Graphics object that will be used for drawing operations
062        */
063       @Override
064       protected void paintComponent(Graphics arg)
065       {
066          super.paintComponent(arg);
067          Graphics2D g2 = (Graphics2D)arg;
068          if (model != null)
069          {
070             g2.setPaint(super.paints.getPegInvalid());
071             int cx, cy;
072             for (Point p : model.illegalPegs())
073             {
074                cx = p.x*csm.getCellWidth()-csm.getWallWidthHalf();
075                cy = p.y*csm.getCellHeight()-csm.getWallHeightHalf();
076                g2.fillRect(cx, cy, csm.getWallWidth(), csm.getWallHeight());
077             }
078             if (model.isCenterLegal())
079             {
080                g2.setColor(Color.GREEN);
081                Dimension size = model.getSize();
082                cx = size.width/2*csm.getCellWidth()-csm.getWallWidthHalf();
083                cy = size.height/2*csm.getCellHeight()-csm.getWallHeightHalf();
084                g2.fillRect(cx, cy, csm.getWallWidth(), csm.getWallHeight());
085             }
086    
087             if (mCurrentTemplate != null)
088                mCurrentTemplate.draw((Graphics2D)arg, getCellSize());
089          }
090       }
091    
092       /**
093        * Sets the currently selected template to mt
094        * @param mt the new current template that will be used. May be null.
095        */
096       public void setTemplate(MazeTemplate mt)
097       {
098          mCurrentTemplate = mt;
099       }
100    
101       /**
102        * Applied the currently selected template either setting walls or clearing
103        * walls based on the value of setWall.  If setWall is true, the walls under
104        * the template will be set, otherwise they will be cleared.
105        * @param setWall true if the walls under the template should be set, false
106        *                     if the walls under the template should be cleared
107        */
108       public void applyTemplate(boolean setWall)
109       {
110          if (model == null || mCurrentTemplate == null)
111             return;
112          TemplatePeg[] tp = mCurrentTemplate.getCenterPegs();
113          Point[] cp = mCurrentTemplate.getCenterPoints(getCellSize());
114          Vector<MazeWall> walls = new Vector<MazeWall>();
115          TreeSet<TemplatePeg> applied = new TreeSet<TemplatePeg>();
116          for (int i = 0; i < Math.min(tp.length, cp.length); i++)
117          {
118             if (applied.contains(tp[i]))
119                continue;
120    
121             MazeCell hostCell;
122             try
123             {
124                hostCell = this.getHostMazeCell(cp[i]);
125             }
126             catch (Exception ex)
127             {
128                continue;
129             }
130    
131             int[] coords = {hostCell.getXZeroBased(), hostCell.getYZeroBased()};
132    
133             int leftX = coords[0]*csm.getCellWidth();
134             int leftY = coords[1]*csm.getCellHeight();
135    
136             if (cp[i].x > leftX + csm.getCellWidthHalf()/2 &&
137                 cp[i].x <= leftX + csm.getCellWidthHalf())
138                continue;
139             if (cp[i].y > leftY + csm.getCellHeightHalf()/2 &&
140                 cp[i].y <= leftY + csm.getCellHeightHalf())
141                continue;
142    
143             int halfX = leftX + csm.getCellWidthHalf();
144             if (cp[i].x < halfX)
145                coords[0]--;
146    
147             int halfY = leftY + csm.getCellHeightHalf();
148             if (cp[i].y < halfY)
149                coords[1]--;
150    
151             TreeSet<TemplatePeg> visited = new TreeSet<TemplatePeg>();
152    
153             applyPeg(tp[i], visited, walls, coords);
154             applied.addAll(visited);
155          }
156          for (MazeWall ms : walls)
157                ms.set(setWall);
158       }
159    
160       /**
161        * Recursively fetches all walls from the model that will be affected by
162        * the template application.  Each peg is only visited once so that this
163        * method will not recurse infinitely.
164        * @param peg  the peg from which to start.
165        * @param visited a set of all pegs that have already be visited
166        * @param walls a Vector holding all walls that have already been found to be
167        *              affected by the template.  New walls will be added when
168        *              found.
169        * @param coords the maze coordinates for peg which may be outside of the
170        *               actual maze.
171        */
172       private void applyPeg(TemplatePeg peg, TreeSet<TemplatePeg> visited,
173                             Vector<MazeWall> walls, int[] coords)
174       {
175          if (visited.contains(peg))
176             return;
177    
178          visited.add(peg);
179    
180          if (peg.top != null)
181          {
182             if (coords[0] >= 0 && coords[0] < model.getSize().width &&
183                 coords[1] >= 0 && coords[1] < model.getSize().height)
184                walls.add(model.getWall(new MazeCell(coords[0]+1, coords[1]+1),
185                                        Direction.East));
186             
187             int[] newCoords = {coords[0], coords[1]-1};
188             applyPeg(peg.top.mRightTop, visited, walls, newCoords);
189          }
190          if (peg.left != null)
191          {
192             
193             if (coords[0] >= 0 && coords[0] < model.getSize().width &&
194                 coords[1] >= 0 && coords[1] < model.getSize().height)
195                walls.add(model.getWall(new MazeCell(coords[0]+1, coords[1]+1),
196                                        Direction.South));
197    
198             int[] newCoords = {coords[0]-1, coords[1]};
199             applyPeg(peg.left.mLeftBottom, visited, walls, newCoords);
200          }
201          if (peg.right != null)
202          {
203             int[] newCoords = {coords[0]+1, coords[1]};
204             applyPeg(peg.right.mRightTop, visited, walls, newCoords);
205          }
206          if (peg.bottom != null)
207          {
208             int[] newCoords = {coords[0], coords[1]+1};
209             applyPeg(peg.bottom.mLeftBottom, visited, walls, newCoords);
210          }
211    
212       }
213    }
