001    package maze.gui.mazeeditor;
002    
003    import java.awt.Color;
004    import java.awt.Dimension;
005    import java.awt.Graphics;
006    import java.awt.Graphics2D;
007    import java.awt.Point;
008    import java.awt.event.ComponentEvent;
009    import java.util.TreeSet;
010    import java.util.Vector;
011    
012    import maze.model.CellSizeModel;
013    import maze.model.Direction;
014    import maze.model.MazeCell;
015    import maze.model.MazeCellPeg;
016    import maze.model.MazeModel.MazeWall;
017    
018    /**
019     * This class provides a MazeView on which maze templates can be applied to the
020     * underlying model. This view is responsible for drawing the current template
021     * that will be applied.
022     * @author Johnathan Smith
023     */
024    public final class EditableMazeView extends maze.gui.MazeView
025    {
026       private static final int WALL_SIZE_DIVIDER = 3;
027       private MazeTemplate mCurrentTemplate = null;
028    
029       /**
030        * Resizes the wall width and height of the maze based on the size of this
031        * MazeView panel when the panel is resized.
032        * @param e the ComponentEvent passed when this component is resized
033        */
034       @Override
035       public void componentResized(ComponentEvent e)
036       {
037          super.componentResized(e);
038          //We override the standard wall size here so it is easier to click on.
039          final int wallSize = Math.min(csm.getCellWidth(), csm.getCellHeight()) / WALL_SIZE_DIVIDER;
040          this.csm.setWallWidth(wallSize);
041          this.csm.setWallHeight(wallSize);
042       }
043    
044       /**
045        * Get the cell size model.
046        * @return A copy of the model instance.
047        */
048       public CellSizeModel getCellSizeModel()
049       {
050          return this.csm.clone();
051       }
052    
053       /**
054        * Paints the maze with a call to MazeView.paintComponent and paints the
055        * currently selected maze template if one is selected.
056        * @param arg the Graphics object that will be used for drawing operations
057        */
058       @Override
059       protected void paintComponent(Graphics arg)
060       {
061          super.paintComponent(arg);
062          final Graphics2D g2 = (Graphics2D) arg;
063          if (model != null)
064          {
065             g2.setPaint(super.paints.getPegInvalid());
066             int cx, cy;
067             for (MazeCellPeg p : model.illegalPegs())
068             {
069                cx = p.getX() * csm.getCellWidth() - csm.getWallWidthHalf();
070                cy = p.getY() * csm.getCellHeight() - csm.getWallHeightHalf();
071                g2.fillRect(cx, cy, csm.getWallWidth(), csm.getWallHeight());
072             }
073             if (model.isCenterLegal())
074             {
075                g2.setColor(Color.GREEN);
076                Dimension size = model.getSize();
077                cx = size.width / 2 * csm.getCellWidth() - csm.getWallWidthHalf();
078                cy = size.height / 2 * csm.getCellHeight() - csm.getWallHeightHalf();
079                g2.fillRect(cx, cy, csm.getWallWidth(), csm.getWallHeight());
080             }
081    
082             if (mCurrentTemplate != null)
083                mCurrentTemplate.draw(g2, this.getCellSizeModel(), this.paints);
084          }
085       }
086    
087       /**
088        * Sets the currently selected template to mt
089        * @param mt the new current template that will be used. May be null.
090        */
091       public void setTemplate(MazeTemplate mt)
092       {
093          mCurrentTemplate = mt;
094       }
095    
096       /**
097        * Applied the currently selected template either setting walls or clearing
098        * walls based on the value of setWall. If setWall is true, the walls under
099        * the template will be set, otherwise they will be cleared.
100        * @param setWall true if the walls under the template should be set, false
101        *           if the walls under the template should be cleared
102        */
103       public void applyTemplate(boolean setWall)
104       {
105          if (model == null || mCurrentTemplate == null)
106             return;
107          TemplatePeg[] tp = mCurrentTemplate.getCenterPegs();
108          Point[] cp = mCurrentTemplate.getCenterPoints(this.getCellSizeModel());
109          Vector<MazeWall> walls = new Vector<MazeWall>();
110          TreeSet<TemplatePeg> applied = new TreeSet<TemplatePeg>();
111          for (int i = 0; i < Math.min(tp.length, cp.length); i++)
112          {
113             if (applied.contains(tp[i]))
114                continue;
115    
116             MazeCell hostCell;
117             try
118             {
119                hostCell = this.getHostMazeCell(cp[i]);
120             }
121             catch (Exception ex)
122             {
123                continue;
124             }
125    
126             int[] coords =
127             {
128                hostCell.getXZeroBased(), hostCell.getYZeroBased()
129             };
130    
131             int leftX = coords[0] * csm.getCellWidth();
132             int leftY = coords[1] * csm.getCellHeight();
133    
134             if (cp[i].x > leftX + csm.getCellWidthHalf() / 2 &&
135                 cp[i].x <= leftX + csm.getCellWidthHalf())
136                continue;
137             if (cp[i].y > leftY + csm.getCellHeightHalf() / 2 &&
138                 cp[i].y <= leftY + csm.getCellHeightHalf())
139                continue;
140    
141             int halfX = leftX + csm.getCellWidthHalf();
142             if (cp[i].x < halfX)
143                coords[0]--;
144    
145             int halfY = leftY + csm.getCellHeightHalf();
146             if (cp[i].y < halfY)
147                coords[1]--;
148    
149             TreeSet<TemplatePeg> visited = new TreeSet<TemplatePeg>();
150    
151             applyPeg(tp[i], visited, walls, coords);
152             applied.addAll(visited);
153          }
154          for (MazeWall ms : walls)
155             ms.set(setWall);
156       }
157    
158       /**
159        * Recursively fetches all walls from the model that will be affected by the
160        * template application. Each peg is only visited once so that this method
161        * will not recurse infinitely.
162        * @param peg the peg from which to start.
163        * @param visited a set of all pegs that have already be visited
164        * @param walls a Vector holding all walls that have already been found to be
165        *           affected by the template. New walls will be added when found.
166        * @param coords the maze coordinates for peg which may be outside of the
167        *           actual maze.
168        */
169       private void applyPeg(TemplatePeg peg, TreeSet<TemplatePeg> visited, Vector<MazeWall> walls,
170             int[] coords)
171       {
172          if (visited.contains(peg))
173             return;
174    
175          visited.add(peg);
176    
177          if (peg.top != null)
178          {
179             if (coords[0] >= 0 &&
180                 coords[0] < model.getSize().width &&
181                 coords[1] >= 0 &&
182                 coords[1] < model.getSize().height)
183                walls.add(model.getWall(new MazeCell(coords[0] + 1, coords[1] + 1), Direction.East));
184    
185             int[] newCoords =
186             {
187                coords[0], coords[1] - 1
188             };
189             applyPeg(peg.top.mRightTop, visited, walls, newCoords);
190          }
191          if (peg.left != null)
192          {
193    
194             if (coords[0] >= 0 &&
195                 coords[0] < model.getSize().width &&
196                 coords[1] >= 0 &&
197                 coords[1] < model.getSize().height)
198                walls.add(model.getWall(new MazeCell(coords[0] + 1, coords[1] + 1), Direction.South));
199    
200             int[] newCoords =
201             {
202                coords[0] - 1, coords[1]
203             };
204             applyPeg(peg.left.mLeftBottom, visited, walls, newCoords);
205          }
206          if (peg.right != null)
207          {
208             int[] newCoords =
209             {
210                coords[0] + 1, coords[1]
211             };
212             applyPeg(peg.right.mRightTop, visited, walls, newCoords);
213          }
214          if (peg.bottom != null)
215          {
216             int[] newCoords =
217             {
218                coords[0], coords[1] + 1
219             };
220             applyPeg(peg.bottom.mLeftBottom, visited, walls, newCoords);
221          }
222    
223       }
224    }
