020    /**
021     * This class provides a MazeView on which maze templates can be applied to the
022     * underlying model. This view is responsible for drawing the current template
023     * that will be applied.
024     * @author Johnathan Smith
025     */
026    public final class EditableMazeView extends MazeView
027    {
028       private static final int WALL_SIZE_DIVIDER = 3;
029       private MazeCell activeCell;
030       private Direction activeWallDir;
031       private MazeTemplate mCurrentTemplate = null;
032    
033       private final transient MouseAdapter mouseAdapter = new MouseAdapter()
034       {
035          @Override
036          public void mouseDragged(MouseEvent e)
037          {
038             if (model != null)
039             {
040                setActiveWall(null, null);
041                final MazeCell cell = getHostMazeCell(e.getPoint());
042                if (cell != null)
043                {
044                   final MazeWall wall = getWall(e.getPoint());
045                   if (wall != null)
046                   {
047                      if (SwingUtilities.isLeftMouseButton(e))
048                      {
049                         if (wall.isSet() == false)
050                         {
051                            wall.set(true);
052                            invalidateCell(cell);
053                         }
054                      }
055                      else if (SwingUtilities.isRightMouseButton(e))
056                      {
057                         if (wall.isSet())
058                         {
059                            wall.set(false);
060                            invalidateCell(cell);
061                         }
062                      }
063                   }
064                }
065             }
066          }
067    
068          @Override
069          public void mouseMoved(MouseEvent e)
070          {
071             if (model != null)
072             {
073                MazeCell newActive = getHostMazeCell(e.getPoint());
074                if (newActive != null)
075                {
076                   if (getWallArea(newActive, Direction.East).contains(e.getPoint()) &&
077                       newActive.getX() < model.getSize().width)
078                   {
079                      setActiveWall(newActive, Direction.East);
080                   }
081                   else if (getWallArea(newActive, Direction.South).contains(e.getPoint()) &&
082                            newActive.getY() < model.getSize().height)
083                   {
084                      setActiveWall(newActive, Direction.South);
085                   }
086                   else
087                   {
088                      setActiveWall(null, null);
089                   }
090                }
091                else
092                {
093                   setActiveWall(null, null);
094                }
095             }
096          }
097    
098          @Override
099          public void mousePressed(MouseEvent e)
100          {
101             if (model != null)
102             {
103                final MazeWall wall = getWall(e.getPoint());
104                if (wall != null)
105                {
106                   //Flip the status of the wall.
107                   wall.set(!wall.isSet());
108                }
109             }
110          }
111       };

113       /**
114        * Constructor.
115        */
116       public EditableMazeView()
117       {
118          super.wallSizeDivider = WALL_SIZE_DIVIDER;
119          this.setEditable(true);
120       }

122       /**
123        * Recursively fetches all walls from the model that will be affected by the
124        * template application. Each peg is only visited once so that this method
125        * will not recurse infinitely.
126        * @param peg the peg from which to start.
127        * @param visited a set of all pegs that have already be visited
128        * @param walls a Vector holding all walls that have already been found to be
129        *           affected by the template. New walls will be added when found.
130        * @param coords the maze coordinates for peg which may be outside of the
131        *           actual maze.
132        */
133       private void applyPeg(TemplatePeg peg, TreeSet<TemplatePeg> visited, Vector<MazeWall> walls,
134             int[] coords)
135       {
136          if (visited.contains(peg))
137             return;
138    
139          visited.add(peg);
140    
141          if (peg.top != null)
142          {
143             if (coords[0] >= 0 &&
144                 coords[0] < model.getSize().width &&
145                 coords[1] >= 0 &&
146                 coords[1] < model.getSize().height)
147                walls.add(model.getWall(new MazeCell(coords[0] + 1, coords[1] + 1), Direction.East));
148    
149             int[] newCoords =
150             {
151                coords[0], coords[1] - 1
152             };
153             applyPeg(peg.top.mRightTop, visited, walls, newCoords);
154          }
155          if (peg.left != null)
156          {
157    
158             if (coords[0] >= 0 &&
159                 coords[0] < model.getSize().width &&
160                 coords[1] >= 0 &&
161                 coords[1] < model.getSize().height)
162                walls.add(model.getWall(new MazeCell(coords[0] + 1, coords[1] + 1), Direction.South));
163    
164             int[] newCoords =
165             {
166                coords[0] - 1, coords[1]
167             };
168             applyPeg(peg.left.mLeftBottom, visited, walls, newCoords);
169          }
170          if (peg.right != null)
171          {
172             int[] newCoords =
173             {
174                coords[0] + 1, coords[1]
175             };
176             applyPeg(peg.right.mRightTop, visited, walls, newCoords);
177          }
178          if (peg.bottom != null)
179          {
180             int[] newCoords =
181             {
182                coords[0], coords[1] + 1
183             };
184             applyPeg(peg.bottom.mLeftBottom, visited, walls, newCoords);
185          }
186       }

188       /**
189        * Applied the currently selected template either setting walls or clearing
190        * walls based on the value of setWall. If setWall is true, the walls under
191        * the template will be set, otherwise they will be cleared.
192        * @param setWall true if the walls under the template should be set, false
193        *           if the walls under the template should be cleared
194        */
195       public void applyTemplate(boolean setWall)
196       {
197          if (model == null || mCurrentTemplate == null)
198             return;
199          TemplatePeg[] tp = mCurrentTemplate.getCenterPegs();
200          Point[] cp = mCurrentTemplate.getCenterPoints(this.getCellSizeModel());
201          Vector<MazeWall> walls = new Vector<MazeWall>();
202          TreeSet<TemplatePeg> applied = new TreeSet<TemplatePeg>();
203          for (int i = 0; i < Math.min(tp.length, cp.length); i++)
204          {
205             if (applied.contains(tp[i]))
206                continue;
207    
208             MazeCell hostCell = this.getHostMazeCell(cp[i]);
209             if (hostCell != null)
210             {
211                int[] coords =
212                {
213                   hostCell.getXZeroBased(), hostCell.getYZeroBased()
214                };
215    
216                int leftX = coords[0] * csm.getCellWidth();
217                int leftY = coords[1] * csm.getCellHeight();
218    
219                if (cp[i].x > leftX + csm.getCellWidthHalf() / 2 &&
220                    cp[i].x <= leftX + csm.getCellWidthHalf())
221                   continue;
222                if (cp[i].y > leftY + csm.getCellHeightHalf() / 2 &&
223                    cp[i].y <= leftY + csm.getCellHeightHalf())
224                   continue;
225    
226                int halfX = leftX + csm.getCellWidthHalf();
227                if (cp[i].x < halfX)
228                   coords[0]--;
229    
230                int halfY = leftY + csm.getCellHeightHalf();
231                if (cp[i].y < halfY)
232                   coords[1]--;
233    
234                TreeSet<TemplatePeg> visited = new TreeSet<TemplatePeg>();
235    
236                applyPeg(tp[i], visited, walls, coords);
237                applied.addAll(visited);
238             }
239          }
240          for (MazeWall ms : walls)
241             ms.set(setWall);
242       }

244       /**
245        * Get the cell size model.
246        * @return A copy of the model instance.
247        */
248       public CellSizeModel getCellSizeModel()
249       {
250          return this.csm.clone();
251       }
252    
253       /**
254        * Get the cell that the given point is located in. This gives us a means to
255        * translate between coordinate positions and cells.
256        * @param pointerLocation A coordinate point in the maze area.
257        * @return The cell or null if the point is not inside a valid cell.
258        */
259       protected MazeCell getHostMazeCell(Point pointerLocation)
260       {
261          try
262          {
263             // If the pointer is on the North or West border wall.
264             if (pointerLocation.x < csm.getWallWidth() || pointerLocation.y < csm.getWallHeight())
265                return null;
266             MazeCell cell = new MazeCell( ( (pointerLocation.x - csm.getWallWidth()) / csm.getCellWidth()) + 1,
267                                          ( (pointerLocation.y - csm.getWallHeight()) / csm.getCellHeight()) + 1);
268             if (cell.isInRange(this.model.getSize()))
269                return cell;
270          }
271          catch (IllegalArgumentException e)
272          {}
273          return null;
274       }
275    
276       /**
277        * Converts a mouse pointer location into an actual maze wall object.
278        * @param pointerLocation Cursor location or other point.
279        * @return The maze wall that can be checked or changed, or null if the
280        *         pointer is not over a wall.
281        */
282       protected MazeWall getWall(Point pointerLocation)
283       {
284          final MazeCell cell = this.getHostMazeCell(pointerLocation);
285          if (cell != null)
286          {
287             if (super.getWallArea(cell, Direction.East).contains(pointerLocation))
288                return super.model.getWall(cell, Direction.East);
289             if (super.getWallArea(cell, Direction.South).contains(pointerLocation))
290                return super.model.getWall(cell, Direction.South);
291          }
292          return null;
293       }

295       /**
296        * Paints the currently selected maze template if one is selected. Also
297        * paints valid and invalid pegs.
298        * @param arg the Graphics object that will be used for drawing operations.
299        */
300       @Override
301       protected void paintComponent(Graphics arg)
302       {
303          super.paintComponent(arg);
304          final Graphics2D g2 = (Graphics2D) arg;
305          if (model != null)
306          {
307             //TODO: Pegs can be drawn in the base maze view class with better performance.
308             for (MazeCellPeg p : model.illegalPegs())
309             {
310                super.painter.drawPegInvalid(g2, super.getPegArea(p.getMazeCell()));
311             }
312             if (model.isCenterLegal())
313                super.painter.drawPegValid(g2, super.getPegArea(model.getWinningCells()[0]));
314    
315             if (mCurrentTemplate != null)
316                mCurrentTemplate.draw(g2, this.getCellSizeModel(), this.painter);
317          }
318          if (this.activeCell != null && this.activeWallDir != null)
319          {
320             super.painter.drawCellHover(g2, super.getWallArea(this.activeCell, this.activeWallDir));
321          }
322       }
323    
324       /**
325        * The wall that the mouse is hovering over should be set active so it can be
326        * highlighted.
327        * @param newCell Cell containing the active wall.
328        * @param newDir Which wall is active.
329        */
330       private void setActiveWall(MazeCell newCell, Direction newDir)
331       {
332          boolean repaint = false;
333          if ( (newCell == null && activeCell != null) ||
334              (newCell != null && newCell.equals(activeCell) == false))
335          {
336             activeCell = newCell;
337             repaint = true;
338          }
339          if (this.activeWallDir != newDir)
340          {
341             this.activeWallDir = newDir;
342             repaint = true;
343          }
344          if (repaint)
345             this.repaint();
346       }
347    
348       /**
349        * Sets whether this MazeView allows mouse interactions to modify its
350        * underlying MazeModel.
351        * @param editable true to allow editing.
352        */
353       public void setEditable(boolean editable)
354       {
355          if (editable)
356          {
357             this.addMouseListener(this.mouseAdapter);
358             this.addMouseMotionListener(this.mouseAdapter);
359          }
360          else
361          {
362             this.removeMouseListener(this.mouseAdapter);
363             this.removeMouseMotionListener(this.mouseAdapter);
364             this.activeCell = null;
365          }
366       }
367    
368       /**
369        * Sets the currently selected template to mt
370        * @param mt the new current template that will be used. May be null.
371        */
372       public void setTemplate(MazeTemplate mt)
373       {
374          mCurrentTemplate = mt;
375       }
