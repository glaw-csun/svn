034    public class NewMazeDialog extends JDialog
035    {
036       public static final String MAZ = "maz";
037       public static final String MZ2 = "mz2";
038       private JTextField mName;
039       private JSpinner mWidth, mHeight;
040       private ButtonGroup mGroup;
041       private JRadioButton mOldSelect, mNewSelect;
042       private JLabel mWidthLabel, mHeightLabel, mNameLabel;
043       private String mSelection = null;
044    
045       public NewMazeDialog(Frame owner)
046       {
047          super(owner, "New Maze", true);
048    
049          JPanel mainPanel = new JPanel(new GridBagLayout());
050    
051          // Create Button Group and radio buttons
052          mGroup = new ButtonGroup();
053          mOldSelect = new JRadioButton(".maz (A standard 16x16 with no name)");
054          mOldSelect.setActionCommand(".maz");
055          ActionListener l = new RadioButtonAction(mOldSelect);
056          mOldSelect.addActionListener(l);
057          mNewSelect = new JRadioButton(".mz2 (Custom Size and Name)");
058          mNewSelect.setActionCommand(".mz2");
059          mNewSelect.addActionListener(l);
060          mGroup.add(mOldSelect);
061          mGroup.setSelected(mOldSelect.getModel(), true);
062          mGroup.add(mNewSelect);
063    
064          // Add radio buttons
065          GridBagConstraints gbc = new GridBagConstraints();
066          gbc.ipadx = 0;
067          gbc.gridx = gbc.gridy = 0;
068          gbc.gridwidth = 2;
069          gbc.anchor = GridBagConstraints.WEST;
070          mainPanel.add(mOldSelect, gbc);
071          gbc.gridy++;
072          mainPanel.add(mNewSelect, gbc);
073    
074          // Create and add the labels for the maze size spinners
075          mWidthLabel = new JLabel("Width");
076          mHeightLabel = new JLabel("Height");
077          gbc.gridy = 2;
078          gbc.gridwidth = 1;
079          mainPanel.add(mWidthLabel, gbc);
080          gbc.gridx = 1;
081          mainPanel.add(mHeightLabel, gbc);
082    
083          // Create and add the maze size spinners
084          mWidth = new JSpinner(new MazeSizeSpinnerModel());
085          mHeight = new JSpinner(new MazeSizeSpinnerModel());
086          mWidth.setFocusable(false);
087          mWidth.setRequestFocusEnabled(false);
088          mHeight.setFocusable(false);
089          mHeight.setRequestFocusEnabled(false);
090          gbc.gridx--;
091          gbc.gridy++;
092          gbc.fill = GridBagConstraints.HORIZONTAL;
093          gbc.weightx = .5;
094          mainPanel.add(mWidth, gbc);
095          gbc.gridx++;
096          mainPanel.add(mHeight, gbc);
097          gbc.weightx = 0;
098          gbc.fill = GridBagConstraints.NONE;
099    
100          // Create and add name label
101          mNameLabel = new JLabel("Name");
102          gbc.gridy++;
103          gbc.gridx--;
104          gbc.gridwidth = 2;
105          mainPanel.add(mNameLabel, gbc);
106    
107          // Create and add name text field
108          mName = new JTextField();
109          mName.addFocusListener(new FocusAdapter()
110          {
111             @Override
112             public void focusGained(FocusEvent e)
113             {
114                super.focusGained(e);
115                mName.setBackground(UIManager.getColor("Panel.background"));
116             }
117          });
118          gbc.fill = GridBagConstraints.HORIZONTAL;
119          gbc.gridy++;
120          mainPanel.add(mName, gbc);
121          gbc.fill = GridBagConstraints.NONE;
122    
123          // Create listener for OK button and text field Enter action
124          l = new ActionListener()
125          {
126             @Override
127             public void actionPerformed(ActionEvent e)
128             {
129                if (mNewSelect.isSelected())
130                {
131                   if (mName.getText() == null || mName.getText().equals(""))
132                      mName.setBackground(Color.PINK);
133                   else
134                   {
135                      mSelection = MZ2;
136                      setVisible(false);
137                   }
138                }
139                else
140                {
141                   mSelection = MAZ;
142                   setVisible(false);
143                }
144             }
145          };
146    
147          // Add ActionListener to name field
148          mName.addActionListener(l);
149    
150          // Create Ok button, add it and add the ActionListener to it
151          JButton ok = new JButton("Ok");
152          ok.addActionListener(l);
153          gbc.gridy++;
154          gbc.anchor = GridBagConstraints.CENTER;
155          mainPanel.add(ok, gbc);
156    
157          // Set a border for the main panel and it to the dialog
158          mainPanel.setBorder(new EmptyBorder(5,5,5,5));
159          add(mainPanel);
160    
161          setResizable(false);
162          setOpsEnabled(false);
163    
164          pack();
165          Point loc = owner.getLocation();
166          Dimension size = owner.getSize();
167          this.setBounds(loc.x+size.width/2-this.getWidth()/2,
168                         loc.y+size.height/2-this.getHeight()/2,
169                         getWidth(), getHeight());
170       }
171    
172       public String showDialog()
173       {
174          mSelection = null;
175          setVisible(true);
176          return mSelection;
177       }
178    
179       public String getText()
180       {
181          return mName.getText();
182       }
183    
184       public Dimension getMazeSize()
185       {
186          return new Dimension((Integer)mWidth.getValue(),
187                               (Integer)mHeight.getValue());
188       }
189    
190       private void setOpsEnabled(boolean enabled)
191       {
192          mHeight.setEnabled(enabled);
193          mWidth.setEnabled(enabled);
194          mName.setEnabled(enabled);
195          mWidthLabel.setEnabled(enabled);
196          mHeightLabel.setEnabled(enabled);
197          mName.setEnabled(enabled);
198          mNameLabel.setEnabled(enabled);
199          mName.requestFocus();
200       }
201    
202       class RadioButtonAction implements ActionListener
203       {
204          private JRadioButton last;
205          public RadioButtonAction(JRadioButton last)
206          {
207             this.last = last;
208          }
209          
210          @Override
211          public void actionPerformed(ActionEvent e)
212          {
213             if (e.getSource().equals(last))
214                return;
215             if (e.getActionCommand().equals(".maz"))
216                setOpsEnabled(false);
217             else
218                setOpsEnabled(true);
219             last = (JRadioButton)e.getSource();
220          }
221    
222       }
223    }

225    class MazeSizeSpinnerModel extends AbstractSpinnerModel
226    {
227       private int mCurrentValue = 16;
228       @Override
229       public Object getValue()
230       {
231          return mCurrentValue;
232       }
233    
234       @Override
235       public void setValue(Object value)
236       {
237          int val;
238          if (value instanceof Integer)
239             val = (Integer)value;
240          else if (value instanceof String)
241          {
242             try
243             {
244                val = Integer.parseInt((String)value);
245             }
246             catch (NumberFormatException ex)
247             {
248                return;
249             }
250          }
251          else
252             return;
253    
254          if (val <= 0)
255             return;
256    
257          if (val%2 == 1)
258             val++;
259          mCurrentValue = val;
260       }
261    
262       @Override
263       public Object getNextValue()
264       {
265          mCurrentValue+=2;
266          fireStateChanged();
267          return mCurrentValue;
268       }
269    
270       @Override
271       public Object getPreviousValue()
272       {
273          mCurrentValue = Math.max(mCurrentValue-2, 4);
274          fireStateChanged();
275          return mCurrentValue;
276       }
277    }
