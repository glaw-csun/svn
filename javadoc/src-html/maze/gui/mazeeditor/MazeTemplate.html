001    package maze.gui.mazeeditor;
002    
003    import java.awt.Color;
004    import java.awt.GradientPaint;
005    import java.awt.Graphics2D;
006    import java.awt.Paint;
007    import java.awt.Point;
008    
009    import java.util.TreeSet;
010    import javax.swing.ImageIcon;
011    
012    import maze.gui.CellSize;
013    
014    /**
015     *
016     * @author John Smith
017     */
018    public abstract class MazeTemplate
019    {
020       protected Paint mWallPaint;
021       protected static final Color PEG_COLOR = Color.BLACK;
022       protected ImageIcon mIcon = null;
023       protected String mDesc = "";
024    
025       public abstract TemplatePeg[] getCenterPegs();
026       public abstract Point[] getCenterPoints(CellSize size);
027       public abstract void updatePosition(Point p, CellSize size);
028       public abstract void nextOrientation();
029       public abstract void grow();
030       public abstract void shrink();
031       public void draw(Graphics2D g, CellSize size)
032       {
033          mWallPaint = new GradientPaint(new Point(0, 0), new Color(0, 94, 189),
034                                         new Point(size.getTotalWidth()/2, 
035                                         size.getTotalHeight()/2),
036                                         new Color(0, 56, 112), true);
037    
038          TemplatePeg[] pegs = this.getCenterPegs();
039          Point[] points = this.getCenterPoints(size);
040    
041          TreeSet<TemplatePeg> visited = new TreeSet<TemplatePeg>();
042          for (int i = 0; i < points.length; i+=2)
043          {
044             Point origin = new Point(points[i].x-size.getWallWidth(),
045                                      points[i].y-size.getWallHeightHalf());
046    
047             drawPeg(pegs[i],visited, g, size, origin);
048          }
049       }
050    
051       public abstract void reset();
052       
053       public ImageIcon getTemplateIcon()
054       {
055          return mIcon;
056       }
057       public String getTemplateDescription()
058       {
059          return mDesc;
060       }
061    
062       protected void drawPeg(TemplatePeg peg, TreeSet<TemplatePeg> visited,
063                              Graphics2D g, CellSize size, Point origin)
064       {
065          if (peg == null || visited.contains(peg))
066             return;
067    
068          visited.add(peg);
069    
070          g.setPaint(null);
071          g.setColor(PEG_COLOR);
072          g.fillRect(origin.x, origin.y, size.getWallWidth(), size.getWallHeight());
073    
074          if (peg.top != null)
075          {
076             origin.y -= size.getCellHeight();
077             g.setPaint(mWallPaint);
078             g.fillRect(origin.x, origin.y, size.getWallWidth(),
079                        size.getCellHeight());
080             origin.y -= size.getWallHeight();
081             drawPeg(peg.top.mRightTop, visited, g, size, origin);
082             origin.y += size.getCellHeight()+size.getWallHeight();
083          }
084    
085          if (peg.bottom != null)
086          {
087             origin.y += size.getCellHeight()+size.getWallHeight();
088             drawPeg(peg.bottom.mLeftBottom, visited, g, size, origin);
089             origin.y -= size.getCellHeight()+size.getWallHeight();
090          }
091    
092          if (peg.left != null)
093          {
094             origin.x -= size.getCellWidth();
095             g.setPaint(mWallPaint);
096             g.fillRect(origin.x, origin.y, size.getCellWidth(),
097                        size.getWallHeight());
098             origin.x -= size.getWallWidth();
099             drawPeg(peg.left.mLeftBottom, visited, g, size, origin);
100             origin.x += size.getCellWidth()+size.getWallWidth();
101          }
102    
103          if (peg.right != null)
104          {
105             origin.x += size.getCellWidth()+size.getWallWidth();
106             drawPeg(peg.right.mRightTop, visited, g, size, origin);
107             origin.x -= size.getCellWidth()+size.getWallWidth();
108          }
109       }
110    }
