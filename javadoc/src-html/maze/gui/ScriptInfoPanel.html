001    package maze.gui;
002    
003    import javax.swing.BorderFactory;
004    import javax.swing.BoxLayout;
005    import javax.swing.DefaultListModel;
006    import javax.swing.JLabel;
007    import javax.swing.JList;
008    import javax.swing.JPanel;
009    import javax.swing.JScrollPane;
010    
011    /**
012     * Creates a panel to display information about the current python script. This
013     * class is a singelton.
014     * @author Luke Last
015     */
016    public final class ScriptInfoPanel extends JPanel
017    {
018       private static ScriptInfoPanel instance;
019       private final JList list = new JList();
020       private final DefaultListModel model = new DefaultListModel();
021       private final JLabel labelErrorStatus = new JLabel();
022       private final JLabel labelSelectedToken = new JLabel();
023       private final JLabel labelTokenValue = new JLabel();
024       private final JLabel labelTokenType = new JLabel();
025    
026       public synchronized static ScriptInfoPanel getInstance()
027       {
028          if (instance == null)
029             instance = new ScriptInfoPanel();
030          return instance;
031       }
032    
033       /**
034        * Constructor, build the panel.
035        */
036       private ScriptInfoPanel()
037       {
038          this.setLayout(new BoxLayout(this, BoxLayout.Y_AXIS));
039          this.list.setModel(this.model);
040          this.add(this.labelErrorStatus);
041          this.add(this.labelSelectedToken);
042          this.add(this.labelTokenType);
043          this.add(this.labelTokenValue);
044          final JScrollPane scrollpane = new JScrollPane(this.list);
045          scrollpane.setBorder(BorderFactory.createTitledBorder("Available Methods"));
046          this.add(scrollpane);
047       }
048    
049       public JList getList()
050       {
051          return list;
052       }
053    
054       public DefaultListModel getListModel()
055       {
056          return this.model;
057       }
058    
059       public void setScriptError(boolean scriptHasError)
060       {
061          if (scriptHasError)
062          {
063             this.labelErrorStatus.setText("<html>Script Status: <b><font color=red>Errors</font></b>");
064          }
065          else
066          {
067             this.labelErrorStatus.setText("<html>Script Status: <b><font color=green>OK</font></b>");
068          }
069       }
070    
071       public void setTokenName(String token)
072       {
073          if (token.isEmpty())
074             this.labelSelectedToken.setText("No token selected");
075          else
076             this.labelSelectedToken.setText("Selected Token: " + token);
077       }
078    
079       public void setTokenEval(String tokenValue)
080       {
081          this.labelTokenValue.setText("Token Eval: " + tokenValue);
082       }
083    
084       public void setTokenType(String tokenType)
085       {
086          this.labelTokenType.setText("Token Type: " + tokenType);
087       }
088    }
