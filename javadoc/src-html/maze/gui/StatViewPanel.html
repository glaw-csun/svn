001    package maze.gui;
002    
003    import java.awt.BorderLayout;
004    import java.awt.event.ActionEvent;
005    import java.awt.event.ActionListener;
006    import java.util.ArrayList;
007    
008    import javax.swing.Box;
009    import javax.swing.BoxLayout;
010    import javax.swing.JComboBox;
011    import javax.swing.JPanel;
012    import javax.swing.JScrollPane;
013    import javax.swing.JSplitPane;
014    import javax.swing.JTable;
015    import javax.swing.SwingUtilities;
016    import javax.swing.event.ListSelectionEvent;
017    import javax.swing.event.ListSelectionListener;
018    import javax.swing.table.DefaultTableModel;
019    
020    import maze.ai.Floodfill;
021    import maze.ai.LeftWallFollower;
022    import maze.ai.RightWallFollower;
023    import maze.ai.RobotBase;
024    import maze.ai.Tremaux;
025    import maze.model.Direction;
026    import maze.model.MazeCell;
027    import maze.model.MazeInfo;
028    import maze.model.MazeModel;
029    import maze.model.RobotModelMaster;
030    
031    /**
032     * @author Vincent Frey
033     */
034    public class StatViewPanel extends JPanel
035    {
036       private StatTracker tracker;
037       private MazeModel maze;
038       private RobotBase algorithm;
039    
040       private DefaultTableModel statTableModel;
041    
042       private ArrayList<RobotBase> algorithms;
043       private ArrayList<String> algorithmNames;
044       private JComboBox algorithmCombo;
045    
046       private final MazeView mazeView = new MazeView();
047    
048       /**
049        * This constructor creates the Panel.
050        */
051       public StatViewPanel()
052       {
053          Box selectionBox = new Box(BoxLayout.Y_AXIS);
054    
055          final MazeList mazeList = new MazeList(this.mazeView);
056          selectionBox.add(mazeList);
057          mazeList.getList().getSelectionModel().addListSelectionListener(new ListSelectionListener()
058          {
059             @Override
060             public void valueChanged(ListSelectionEvent e)
061             {
062                try
063                {
064                   Object o = mazeList.getList().getSelectedValue();
065                   MazeInfo mi = (MazeInfo) o;
066                   if (mi != null)
067                   {
068                      maze = new MazeModel(mi.getModel());
069                      tracker.reload(algorithm, new RobotModelMaster(maze,
070                                                                     new MazeCell(1, 16),
071                                                                     Direction.North));
072                   }
073                   displayStats();
074                }
075                catch (RuntimeException e1)
076                {
077                   e1.printStackTrace();
078                }
079             }
080          });
081    
082          mazeList.getList().getSelectionModel().addListSelectionListener(new ListSelectionListener()
083          {
084             @Override
085             public void valueChanged(ListSelectionEvent e)
086             {
087                Object o = mazeList.getList().getSelectedValue();
088                MazeInfo mi = (MazeInfo) o;
089                maze = new MazeModel(mi.getModel());
090                tracker.reload(algorithm, new RobotModelMaster(maze,
091                                                               new MazeCell(1, 16),
092                                                               Direction.North));
093                displayStats();
094             }
095          });
096    
097          //If time change this to incorporate the RobotBase list
098          algorithms = new ArrayList<RobotBase>();
099          algorithms.add(new LeftWallFollower());
100          algorithms.add(new RightWallFollower());
101          algorithms.add(new Tremaux());
102          algorithms.add(new Floodfill());
103    
104          algorithmNames = new ArrayList<String>();
105          algorithmNames.add("Left Wall Follower");
106          algorithmNames.add("Right Wall Follower");
107          algorithmNames.add("Tremaux");
108          algorithmNames.add("Floodfill");
109    
110          ActionListener algorithmChange = new ActionListener()
111          {
112             public void actionPerformed(ActionEvent action)
113             {
114                algorithm = (RobotBase) algorithmCombo.getSelectedItem();
115                tracker.reload(algorithm, new RobotModelMaster(maze,
116                                                               new MazeCell(1, 16),
117                                                               Direction.North));
118                displayStats();
119             }
120          };
121    
122          algorithmCombo = new JComboBox(RobotBase.getRobotListModel());
123          algorithmCombo.addActionListener(algorithmChange);
124    
125          selectionBox.add(algorithmCombo);
126    
127          JPanel rightPanel = new JPanel();
128          JScrollPane leftSide = new JScrollPane(mazeView);
129          final JSplitPane statSplitPane = new JSplitPane(JSplitPane.HORIZONTAL_SPLIT,
130                                                          leftSide,
131                                                          rightPanel);
132          this.setLayout(new BorderLayout());
133          this.add(statSplitPane, BorderLayout.CENTER);
134          statSplitPane.setDividerLocation(.6);
135          statSplitPane.setResizeWeight(.5);
136    
137          rightPanel.setLayout(new BoxLayout(rightPanel, BoxLayout.Y_AXIS));
138          rightPanel.add(selectionBox);
139    
140          statTableModel = new DefaultTableModel();
141    
142          String uniqueString = "Number of Unique Squares Traversed: ";
143          String firstSquares = "Number of Cells Traversed to Reach the Center Once: ";
144          String firstTurns = "Number of Turns Taken: ";
145          String bestSquares = "Number of Cells Traversed on Best Run: ";
146          String bestTurns = "Number of Turns Taken: ";
147          String bestTotal = "Total Number of Cells Traversed to Complete Best Run: ";
148          String bestTotalTurns = "Number of Turns Taken: ";
149          String[] rowHeadings = new String[7];
150          rowHeadings[0] = uniqueString;
151          rowHeadings[1] = firstSquares;
152          rowHeadings[2] = firstTurns;
153          rowHeadings[3] = bestSquares;
154          rowHeadings[4] = bestTurns;
155          rowHeadings[5] = bestTotal;
156          rowHeadings[6] = bestTotalTurns;
157    
158          statTableModel.addColumn("Statistics of Interest", rowHeadings);
159          statTableModel.addColumn("Values");
160    
161          final JTable statTable = new JTable(statTableModel);
162          statTable.setEnabled(false);
163          //statTable.setAutoResizeMode(JTable.AUTO_RESIZE_ALL_COLUMNS);
164    
165          JScrollPane statPane = new JScrollPane(statTable);
166          rightPanel.add(statPane);
167    
168          MazeInfo mi = (MazeInfo) mazeList.getList().getSelectedValue();
169          if (mi != null)
170          {
171             this.maze = new MazeModel(mi.getModel());
172          }
173          else
174          {
175             //Create a dummy maze so the application doesn't crash.
176             this.maze = new MazeModel(16, 16);
177          }
178    
179          this.algorithm = (RobotBase) algorithmCombo.getSelectedItem();
180    
181          this.tracker = new StatTracker(algorithm, new RobotModelMaster(maze,
182                                                                         new MazeCell(1, 16),
183                                                                         Direction.North));
184          displayStats();
185          SwingUtilities.invokeLater(new Runnable()
186          {
187             @Override
188             public void run()
189             {
190                statSplitPane.setDividerLocation(.6);
191                statSplitPane.setResizeWeight(.6);
192                statTable.doLayout();
193             }
194          });
195       }
196    
197       private void displayStats()
198       {
199          //This function displays all of the information for the panel
200          //First lets display the table
201          statTableModel.setValueAt(String.valueOf(tracker.getTotalTraversed()), 0, 1);
202          if (tracker.getFirstRunCells() != StatTracker.USELESS)
203          {
204             statTableModel.setValueAt(String.valueOf(tracker.getFirstRunCells()), 1, 1);
205             statTableModel.setValueAt(String.valueOf(tracker.getFirstRunTurns()), 2, 1);
206             statTableModel.setValueAt(String.valueOf(tracker.getBestRunCells()), 3, 1);
207             statTableModel.setValueAt(String.valueOf(tracker.getBestRunTurns()), 4, 1);
208             statTableModel.setValueAt(String.valueOf(tracker.getThroughBestRunCells()), 5, 1);
209             statTableModel.setValueAt(String.valueOf(tracker.getThroughBestRunTurns()), 6, 1);
210          }
211          else
212          {
213             statTableModel.setValueAt("N/A", 1, 1);
214             statTableModel.setValueAt("N/A", 2, 1);
215             statTableModel.setValueAt("N/A", 3, 1);
216             statTableModel.setValueAt("N/A", 4, 1);
217             statTableModel.setValueAt("N/A", 5, 1);
218             statTableModel.setValueAt("N/A", 6, 1);
219          }
220    
221          //Now lets display the mazeView
222          mazeView.setModel(maze);
223    
224          mazeView.loadUnexplored(tracker.getAllUnexplored());
225    
226          mazeView.loadFirstRun((ArrayList<MazeCell>) tracker.getFirstRun());
227    
228          mazeView.loadBestRun((ArrayList<MazeCell>) tracker.getBestRun());
229    
230          mazeView.setDrawFog(true);
231          mazeView.setDrawFirstRun(true);
232          mazeView.setDrawBestRun(true);
233    
234       }
235    }
