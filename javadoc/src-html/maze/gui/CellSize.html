005    package maze.gui;
006    
007    /**
008     * This model stores the sizes of the cells and wall segments that are drawn
009     * to the screen.
010     */
011    public class CellSize implements Comparable<CellSize>
012    {
013       private int[] sizes = {44, 50, 10, 10, 0, 0};
014    
015       public CellSize(int cw, int ch, int ww, int wh, int tw, int th)
016       {
017          sizes[0] = cw;
018          sizes[1] = ch;
019          sizes[2] = ww;
020          sizes[3] = wh;
021          sizes[4] = tw;
022          sizes[5] = th;
023       }
024    
025       public int getCellWidth()
026       {
027          return sizes[0];
028       }
029    
030       public int getCellHeight()
031       {
032          return sizes[1];
033       }
034    
035       public int getWallWidth()
036       {
037          return sizes[2];
038       }
039    
040       public int getWallHeight()
041       {
042          return sizes[3];
043       }
044    
045       public int getCellWidthHalf()
046       {
047          return sizes[0] / 2;
048       }
049    
050       public int getCellHeightHalf()
051       {
052          return sizes[1] / 2;
053       }
054    
055       public int getWallWidthHalf()
056       {
057          return sizes[2] / 2;
058       }
059    
060       public int getWallHeightHalf()
061       {
062          return sizes[3] / 2;
063       }
064    
065       public int getTotalWidth()
066       {
067          return sizes[4];
068       }
069    
070       public int getTotalHeight()
071       {
072          return sizes[5];
073       }
074    
075       @Override
076       public int compareTo(CellSize o)
077       {
078          for (int i = 0; i < sizes.length; i++)
079             if (sizes[i]-o.sizes[i] != 0)
080                return sizes[i]-o.sizes[i];
081          return 0;
082       }
083    }
