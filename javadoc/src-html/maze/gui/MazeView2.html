136       /**
137        * Calls the draw cell method on every cell in the maze.
138        * @param g
139        */
140       private void drawAllCells(final Graphics2D g)
141       {
142          //Loop through each cell in the maze.
143          for (int x = 1; x <= this.model.getSize().width; x++)
144          {
145             for (int y = 1; y <= this.model.getSize().height; y++)
146             {
147                final MazeCell cell = new MazeCell(x, y);
148                this.drawCell(g, cell);
149             } //End y loop.
150          } //End x loop.
151       }

153       /**
154        * Draws an arrow graphic.
155        * @param g
156        * @param local
157        * @param x
158        * @param y
159        */
160       private void drawArrow(final Graphics2D g, final Direction local, final int x, final int y)
161       {
162          //Draws an arrow in the direction of "local" centered on the point (x,y)
163          if (local.equals(Direction.North))
164          {
165             final int[] ys =
166             {
167                y + this.csm.getCellHeight() * 3 / 8, y, y, y - this.csm.getCellHeight() * 3 / 8, y, y
168             };
169             final int[] xs =
170             {
171                x, x - this.csm.getCellWidth() / 8, x - this.csm.getCellWidth() / 4, x,
172                x + this.csm.getCellWidth() / 4, x + this.csm.getCellWidth() / 8
173             };
174             g.drawPolygon(xs, ys, 6);
175          }
176          if (local.equals(Direction.South))
177          {
178             final int[] ys =
179             {
180                y - this.csm.getCellHeight() * 3 / 8, y, y, y + this.csm.getCellHeight() * 3 / 8, y, y
181             };
182             final int[] xs =
183             {
184                x, x - this.csm.getCellWidth() / 8, x - this.csm.getCellWidth() / 4, x,
185                x + this.csm.getCellWidth() / 4, x + this.csm.getCellWidth() / 8
186             };
187             g.drawPolygon(xs, ys, 6);
188          }
189          if (local.equals(Direction.West))
190          {
191             final int[] xs =
192             {
193                x + this.csm.getCellWidth() * 3 / 8, x, x, x - this.csm.getCellWidth() * 3 / 8, x, x
194             };
195             final int[] ys =
196             {
197                y, y - this.csm.getCellHeight() / 8, y - this.csm.getCellHeight() / 4, y,
198                y + this.csm.getCellHeight() / 4, y + this.csm.getCellHeight() / 8
199             };
200             g.drawPolygon(xs, ys, 6);
201          }
202          if (local.equals(Direction.East))
203          {
204             final int[] xs =
205             {
206                x - this.csm.getCellWidth() * 3 / 8, x, x, x + this.csm.getCellWidth() * 3 / 8, x, x
207             };
208             final int[] ys =
209             {
210                y, y - this.csm.getCellHeight() / 8, y - this.csm.getCellHeight() / 4, y,
211                y + this.csm.getCellHeight() / 4, y + this.csm.getCellHeight() / 8
212             };
213             g.drawPolygon(xs, ys, 6);
214          }
215       }

217       /**
218        * The primary draw method for a cell. This should draw all aspects of a
219        * cell.
220        * @param g Where to draw.
221        * @param cell The cell in question.
222        */
223       private void drawCell(final Graphics2D g, final MazeCell cell)
224       {
225          this.paints.drawCellBackground(g, this.getCellAreaInner(cell));
226    
227          if (this.model.getWall(cell, Direction.East).isSet())
228          {
229             this.paints.drawWallSet(g, this.getWallArea(cell, Direction.East));
230          }
231          else
232          {
233             this.paints.drawWallEmpty(g, this.getWallArea(cell, Direction.East));
234          }
235          if (this.model.getWall(cell, Direction.South).isSet())
236          {
237             this.paints.drawWallSet(g, this.getWallArea(cell, Direction.South));
238          }
239          else
240          {
241             this.paints.drawWallEmpty(g, this.getWallArea(cell, Direction.South));
242          }
243          this.paints.drawPeg(g, this.getPegArea(cell));
244          if (this.robotPathModel != null)
245          {
246             //Draw the fog of war.
247             if (this.drawFog && !this.robotPathModel.hasCellBeenVisited(cell))
248             {
249                final Rectangle area = this.getCellArea(cell);
250                final MazeCell east = cell.neighbor(Direction.East);
251                final MazeCell south = cell.neighbor(Direction.South);
252                if (east.isInRange(this.model.getSize()) &&
253                    this.robotPathModel.hasCellBeenVisited(east))
254                {
255                   area.width -= this.csm.getWallWidth();
256                }
257                if (south.isInRange(this.model.getSize()) &&
258                    this.robotPathModel.hasCellBeenVisited(south))
259                {
260                   area.height -= this.csm.getWallHeight();
261                }
262                this.paints.drawFog(g, area);
263             }
264             // Draw a current path of dots.
265             if (this.drawPathCurrent && this.robotPathModel.getPathRecent().contains(cell))
266             {
267                final EnumSet<Direction> directions = EnumSet.noneOf(Direction.class);
268                for (final Direction dir : this.getAdjacentDirections(cell))
269                {
270                   if (!this.model.getWall(cell, dir).isSet() &&
271                       this.robotPathModel.getPathRecent().contains(cell.neighbor(dir)))
272                   {
273                      directions.add(dir);
274                   }
275                }
276                //this.paints.drawRunCurrent(g, this.getCellAreaInner(cell), directions);
277             }
278          }
279       }

281       /**
282        * Draws all cells that have been invalidated.
283        * @param g Where to draw.
284        */
285       private void drawInvalidatedCells(final Graphics2D g)
286       {
287          synchronized (this.invalidatedCells)
288          {
289             for (final MazeCell cell : this.invalidatedCells)
290             {
291                this.drawCell(g, cell);
292                //System.out.println("Drew cell: " + cell);
293             }
294             this.invalidatedCells.clear();
295          }
296       }

298       /**
299        * Draws the top and left outside walls as these don't fall inside of any
300        * cells.
301        * @param g Where to draw.
302        */
303       private void drawOutsideWalls(final Graphics2D g)
304       {
305    
306          final Rectangle pegArea = new Rectangle(0,
307                                                  0,
308                                                  this.csm.getWallWidth(),
309                                                  this.csm.getWallHeight());
310          final Rectangle wallArea = new Rectangle(this.csm.getWallWidth(),
311                                                   0,
312                                                   this.csm.getCellWidthInner(),
313                                                   this.csm.getWallHeight());
314    
315          for (int i = 0; i <= this.model.getSize().width; i++)
316          {
317             this.paints.drawPeg(g, pegArea);
318             pegArea.x += this.csm.getCellWidth();
319             // We draw more pegs than walls.
320             if (i == 0)
321             {
322                continue;
323             }
324             this.paints.drawWallSet(g, wallArea);
325             wallArea.x += this.csm.getCellWidth();
326          }
327    
328          pegArea.setLocation(0, this.csm.getCellHeight());
329          wallArea.setSize(this.csm.getWallWidth(), this.csm.getCellHeightInner());
330          wallArea.setLocation(0, this.csm.getWallHeight());
331          // Draw the left side column of pegs and walls.
332          for (int i = 1; i <= this.model.getSize().height; i++)
333          {
334             this.paints.drawPeg(g, pegArea);
335             pegArea.y += this.csm.getCellHeight();
336             this.paints.drawWallSet(g, wallArea);
337             wallArea.y += this.csm.getCellHeight();
338          }
339       }

341       /**
342        * Get the cell that the given point is located in. This gives us a means to
343        * translate between coordinate positions and cells.
344        * @param location A coordinate point in the maze area.
345        * @return The cell or null if the point is not inside a valid cell.
346        */
347       @SuppressWarnings("unused")
348       private MazeCell getHostCell(Point location)
349       {
350          try
351          {
352             MazeCell cell = new MazeCell( ( (location.x - csm.getWallWidth()) / csm.getCellWidth()) + 1,
353                                          ( (location.y - csm.getWallHeight()) / csm.getCellHeight()) + 1);
354             if (cell.isInRange(model.getSize()))
355             {
356                return cell;
357             }
358          }
359          catch (Exception e)
360          {}
361          return null;
362       }

364       /**
365        * Draw a robot path onto the maze. The path will have the width of the cell
366        * walls.
367        * @param g Where to draw.
368        * @param path The cell to cell path to draw.
369        * @param offset Shift the path by the given amount so that you can draw
370        *           multiple paths without them overlapping. A value of 0 draws to
371        *           the center of the cell.
372        * @param trimTail If set to true the tail of the path will be trimmed to the
373        *           location of the robot. This allows the current path to finish
374        *           right on the current robot position.
375        */
376       private void drawPath(final Graphics2D g, final List<MazeCell> path, final int offset,
377             boolean trimTail)
378       {
379          if (path != null && !path.isEmpty())
380          {
381             MazeCell here = path.get(0);
382             MazeCell there;
383             int x, y;
384             int width, height;
385             for (int i = 1; i < path.size(); i++)
386             {
387                there = path.get(i);
388                final Point center = this.getCellCenterInner(here);
389                if (here.getX() < there.getX())
390                {
391                   //here is west of there, going east.
392                   x = center.x + this.csm.getWallWidthHalf();
393                   y = center.y - this.csm.getWallHeightHalf();
394                   width = this.csm.getCellWidth();
395                   height = this.csm.getWallHeight();
396                }
397                else if (here.getX() > there.getX())
398                {
399                   //here is east of there, going west.
400                   x = center.x - this.csm.getWallWidthHalf() - this.csm.getCellWidth();
401                   y = center.y - this.csm.getWallHeightHalf();
402                   width = this.csm.getCellWidth();
403                   height = this.csm.getWallHeight();
404                }
405                else if (here.getY() > there.getY())
406                {
407                   //here is south of there, going north.
408                   x = center.x - this.csm.getWallWidthHalf();
409                   y = center.y - this.csm.getWallHeightHalf() - this.csm.getCellHeight();
410                   width = this.csm.getWallWidth();
411                   height = this.csm.getCellHeight();
412                }
413                else
414                {
415                   //here is north of there, going south.
416                   x = center.x - this.csm.getWallWidthHalf();
417                   y = center.y + this.csm.getWallHeightHalf();
418                   width = this.csm.getWallWidth();
419                   height = this.csm.getCellHeight();
420                }
421                // If we are at the last cell and we are trimming the tail up to the robot.
422                if (i == path.size() - 1 && trimTail)
423                {
424                   if (here.getX() < there.getX())
425                   {
426                      //here is west of there, going east.
427                      width = Math.abs(x - this.robotLocation.x);
428                   }
429                   else if (here.getX() > there.getX())
430                   {
431                      //here is east of there, going west.
432                      width -= this.robotLocation.x - x;
433                      x = this.robotLocation.x;
434    
435                   }
436                   else if (here.getY() > there.getY())
437                   {
438                      //here is south of there, going north.
439                      height -= this.robotLocation.y - y;
440                      y = this.robotLocation.y;
441                   }
442                   else
443                   {
444                      //here is north of there, going south.
445                      height = this.robotLocation.y - y;
446                   }
447                }
448                g.fillRect(x - offset, y + offset, width, height);
449                here = there;
450             }
451          }
452       }

454       /**
455        * Draws the very top of the maze view. This is the most frequently called
456        * draw method and is called every time the view is repainted. Because of
457        * this it should run as fast as possible. It is much better to draw in the
458        * <code>drawCell()</code> method.
459        * @param g What to draw on.
460        */
461       private void drawTopLayer(final Graphics2D g)
462       {
463          this.setRenderingQualityLow(g);
464          if (this.robotPathModel != null)
465          {
466             if (this.drawPathFirst)
467             {
468                g.setPaint(this.paints.getRunFirst());
469                this.drawPath(g, this.robotPathModel.getPathFirst(), this.csm.getWallWidth(), false);
470             }
471             if (this.drawPathBest)
472             {
473                g.setPaint(this.paints.getRunBest());
474                this.drawPath(g, this.robotPathModel.getPathBest(), -this.csm.getWallWidth(), false);
475             }
476             if (this.drawPathCurrent)
477             {
478                g.setPaint(this.paints.getRunCurrent());
479                this.drawPath(g, this.robotPathModel.getPathRecent(), 0, true);
480             }
481          }
482          if (this.drawUnderstanding)
483          {
484             this.drawUnderstanding(g);
485          }
486          if (this.getRobotLocation() != null)
487          {
488             this.setRenderingQualityHigh(g);
489             this.paints.drawRobot(g,
490                                   this.getRobotLocation(),
491                                   this.getRobotRotation(),
492                                   this.csm.getCellWidthInner(),
493                                   this.csm.getCellHeightInner());
494          }
495       }

517       /**
518        * Draws the arrows and numbers on the maze.
519        * @param g
520        */
521       private void drawUnderstanding(final Graphics2D g)
522       {
523          MazeCell here;
524          if (understandingInt != null)
525          {
526             int local;
527             for (int i = 1; i <= model.getSize().width; i++)
528             {
529                for (int j = 1; j <= model.getSize().height; j++)
530                {
531                   here = new MazeCell(i, j);
532                   g.setColor(Color.BLACK);
533                   final Point center = this.getCellCenterInner(here);
534                   local = understandingInt[i - 1][j - 1];
535                   g.drawString(String.valueOf(local), center.x - 6, center.y + 2);
536                }
537             }
538          }
539          else if (understandingDir != null)
540          {
541             Direction local;
542             for (int i = 1; i <= model.getSize().width; i++)
543             {
544                for (int j = 1; j <= model.getSize().height; j++)
545                {
546                   here = new MazeCell(i, j);
547                   final Point center = this.getCellCenterInner(here);
548                   g.setColor(Color.BLACK);
549                   local = understandingDir[i - 1][j - 1];
550                   if (local != null)
551                   {
552                      drawArrow(g, local, center.x, center.y);
553                   }
554                }
555             }
556          }
557       }

568       /**
569        * Gets all the cells that are adjacent to the given one.
570        * @param cell
571        * @return
572        */
573       @SuppressWarnings("unused")
574       private MazeCell[] getAdjacentCells(final MazeCell cell)
575       {
576          final Direction[] dirs = this.getAdjacentDirections(cell).toArray(new Direction[0]);
577          final MazeCell[] result = new MazeCell[dirs.length];
578          for (int i = 0; i < dirs.length; i++)
579          {
580             result[i] = cell.neighbor(dirs[i]);
581          }
582          return result;
583       }

585       /**
586        * Gets a direction for each adjacent cell that exists. The only time one
587        * doesn't exists is when the given cell is on the maze edge.
588        * @param cell
589        * @return
590        */
591       private EnumSet<Direction> getAdjacentDirections(final MazeCell cell)
592       {
593          // We start with all and then remove because it is less likely.
594          final EnumSet<Direction> directions = EnumSet.allOf(Direction.class);
595          if (cell.getX() == 1)
596          {
597             directions.remove(Direction.West);
598    
599          }
600          if (cell.getY() == 1)
601          {
602             directions.remove(Direction.North);
603    
604          }
605          if (cell.getX() == this.model.getSize().width)
606          {
607             directions.remove(Direction.East);
608    
609          }
610          if (cell.getY() == this.model.getSize().height)
611          {
612             directions.remove(Direction.South);
613    
614          }
615          return directions;
616       }

618       /**
619        * Get the graphics object that draws onto the background image. All drawing
620        * onto the background image must use this.
621        * @return
622        */
623       private Graphics2D getBackgroundGraphics()
624       {
625          if (this.backgroundImage == null)
626          {
627             // If creating a new background image make sure we paint it.
628             this.repaintAll = true;
629             this.backgroundGraphics = null;
630             this.backgroundImage = new BufferedImage(getWidth(),
631                                                      getHeight(),
632                                                      BufferedImage.TYPE_INT_ARGB);
633          }
634          if (this.backgroundGraphics == null)
635          {
636             this.backgroundGraphics = this.backgroundImage.createGraphics();
637             this.setRenderingQualityHigh(this.backgroundGraphics);
638          }
639          return this.backgroundGraphics;
640       }

738       /**
739        * Get the area of a cell wall.
740        * @param cell The cell in question.
741        * @param wall Which wall do you want. Must be East or South.
742        * @return The absolute coordinates of the area.
743        */
744       private Rectangle getWallArea(final MazeCell cell, final Direction wall)
745       {
746          switch (wall)
747          {
748             case East :
749                return new Rectangle(cell.getX() * this.csm.getCellWidth(),
750                                     this.csm.getWallHeight() +
751                                           cell.getYZeroBased() *
752                                           this.csm.getCellHeight(),
753                                     this.csm.getWallWidth(),
754                                     this.csm.getCellHeightInner());
755             case South :
756                return new Rectangle(this.csm.getWallWidth() +
757                                           cell.getXZeroBased() *
758                                           this.csm.getCellWidth(),
759                                     cell.getY() * this.csm.getCellHeight(),
760                                     this.csm.getCellWidthInner(),
761                                     this.csm.getWallHeight());
762             default :
763                throw new IllegalArgumentException("Non supported direction: " + wall);
764          }
765       }

776       /**
777        * Invalidate a cell and mark it to be redrawn.
778        * @param cell The cell to be redrawn.
779        */
780       protected void invalidateCell(final MazeCell cell)
781       {
782          if (cell != null && this.model != null && cell.isInRange(this.model.getSize()))
783          {
784             synchronized (this.invalidatedCells)
785             {
786                this.invalidatedCells.add(cell);
787             }
788             super.repaint();
789          }
790       }

802       /**
803        * Master starting point for the custom painting. We want to draw as little
804        * as we can to keep performance up.
805        */
806       @Override
807       protected void paintComponent(final Graphics arg)
808       {
809          final Graphics2D bgg = this.getBackgroundGraphics();
810          if (this.repaintAll)
811          {
812             this.repaintAll = false;
813             this.drawOutsideWalls(bgg);
814             this.drawAllCells(bgg);
815             // Because the draw all can take so long we still want to redraw invalidated cells after.
816          }
817          this.drawInvalidatedCells(bgg);
818          final Graphics2D g = (Graphics2D) arg;
819          g.drawImage(this.backgroundImage, null, 0, 0);
820          this.drawTopLayer(g);
821       }
