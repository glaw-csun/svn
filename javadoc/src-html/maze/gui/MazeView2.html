124       @Override

126       {
127          if (model != null)
128          {
129             this.backgroundImage = null;
130             this.repaintAll = true;

132                                   this.model.getSize().width);
133             this.csm.setCellHeight( (this.getHeight() - this.csm.getWallHeight()) /
134                                    this.model.getSize().height);
135    
136             final int wallSize = Math.min(csm.getCellWidth(), csm.getCellHeight()) / WALL_SIZE_DIVIDER;
137             this.csm.setWallWidth(wallSize);
138             this.csm.setWallHeight(wallSize);
139             this.paints.setMazeSize(this.getMazeSize());
140             this.repaint();
141          }
142       }

148       /**
149        * Calls the draw cell method on every cell in the maze.
150        * @param g
151        */
152       private void drawAllCells(final Graphics2D g)
153       {
154          //Loop through each cell in the maze.
155          for (int x = 1; x <= this.model.getSize().width; x++)
156          {
157             for (int y = 1; y <= this.model.getSize().height; y++)
158             {
159                final MazeCell cell = new MazeCell(x, y);
160                this.drawCell(g, cell);
161             } //End y loop.
162          } //End x loop.
163       }

165       private void drawArrow(final Graphics2D g, final Direction local, final int x, final int y)
166       {
167          //Draws an arrow in the direction of "local" centered on the point (x,y)
168          if (local.equals(Direction.North))
169          {
170             final int[] ys =
171             {
172                y + this.csm.getCellHeight() * 3 / 8, y, y, y - this.csm.getCellHeight() * 3 / 8, y, y
173             };
174             final int[] xs =
175             {
176                x, x - this.csm.getCellWidth() / 8, x - this.csm.getCellWidth() / 4, x,
177                x + this.csm.getCellWidth() / 4, x + this.csm.getCellWidth() / 8
178             };
179             g.drawPolygon(xs, ys, 6);
180          }
181          if (local.equals(Direction.South))
182          {
183             final int[] ys =
184             {
185                y - this.csm.getCellHeight() * 3 / 8, y, y, y + this.csm.getCellHeight() * 3 / 8, y, y
186             };
187             final int[] xs =
188             {
189                x, x - this.csm.getCellWidth() / 8, x - this.csm.getCellWidth() / 4, x,
190                x + this.csm.getCellWidth() / 4, x + this.csm.getCellWidth() / 8
191             };
192             g.drawPolygon(xs, ys, 6);
193          }
194          if (local.equals(Direction.West))
195          {
196             final int[] xs =
197             {
198                x + this.csm.getCellWidth() * 3 / 8, x, x, x - this.csm.getCellWidth() * 3 / 8, x, x
199             };
200             final int[] ys =
201             {
202                y, y - this.csm.getCellHeight() / 8, y - this.csm.getCellHeight() / 4, y,
203                y + this.csm.getCellHeight() / 4, y + this.csm.getCellHeight() / 8
204             };
205             g.drawPolygon(xs, ys, 6);
206          }
207          if (local.equals(Direction.East))
208          {
209             final int[] xs =
210             {
211                x - this.csm.getCellWidth() * 3 / 8, x, x, x + this.csm.getCellWidth() * 3 / 8, x, x
212             };
213             final int[] ys =
214             {
215                y, y - this.csm.getCellHeight() / 8, y - this.csm.getCellHeight() / 4, y,
216                y + this.csm.getCellHeight() / 4, y + this.csm.getCellHeight() / 8
217             };
218             g.drawPolygon(xs, ys, 6);
219          }
220       }

222       /**
223        * @param g
224        * @param cell
225        */
226       private void drawCell(final Graphics2D g, final MazeCell cell)
227       {
228          this.paints.drawCellBackground(g, this.getCellAreaInner(cell));
229    
230          if (this.model.getWall(cell, Direction.East).isSet())
231          {
232             this.paints.drawWallSet(g, this.getWallArea(cell, Direction.East));
233          }
234          else
235          {
236             this.paints.drawWallEmpty(g, this.getWallArea(cell, Direction.East));
237          }
238          if (this.model.getWall(cell, Direction.South).isSet())
239          {
240             this.paints.drawWallSet(g, this.getWallArea(cell, Direction.South));
241          }
242          else
243          {
244             this.paints.drawWallEmpty(g, this.getWallArea(cell, Direction.South));
245          }
246          this.paints.drawPeg(g, this.getPegArea(cell));
247          if (this.robotPathModel != null)
248          {
249             //Draw the fog of war.
250             if (this.drawFog && !this.robotPathModel.hasCellBeenVisited(cell))
251             {
252                final Rectangle area = this.getCellArea(cell);
253                final MazeCell east = cell.neighbor(Direction.East);
254                final MazeCell south = cell.neighbor(Direction.South);
255                if (east.isInRange(this.model.getSize()) &&
256                    this.robotPathModel.hasCellBeenVisited(east))
257                {
258                   area.width -= this.csm.getWallWidth();
259                }
260                if (south.isInRange(this.model.getSize()) &&
261                    this.robotPathModel.hasCellBeenVisited(south))
262                {
263                   area.height -= this.csm.getWallHeight();
264                }
265                this.paints.drawFog(g, area);
266             }
267             // Draw a current path of dots.
268             if (this.drawPathCurrent && this.robotPathModel.getPathRecent().contains(cell))
269             {
270                final EnumSet<Direction> directions = EnumSet.noneOf(Direction.class);
271                for (final Direction dir : this.getAdjacentDirections(cell))
272                {
273                   if (!this.model.getWall(cell, dir).isSet() &&
274                       this.robotPathModel.getPathRecent().contains(cell.neighbor(dir)))
275                   {
276                      directions.add(dir);
277                   }
278                }
279                //this.paints.drawRunCurrent(g, this.getCellAreaInner(cell), directions);
280             }
281          }
282       }

300       /**
301        * Draws the top and left outside walls as these don't fall inside of any
302        * cells.
303        * @param g
304        */
305       private void drawOutsideWalls(final Graphics2D g)
306       {
307    
308          final Rectangle pegArea = new Rectangle(0,
309                                                  0,
310                                                  this.csm.getWallWidth(),
311                                                  this.csm.getWallHeight());
312          final Rectangle wallArea = new Rectangle(this.csm.getWallWidth(),
313                                                   0,
314                                                   this.csm.getCellWidthInner(),
315                                                   this.csm.getWallHeight());
316    
317          for (int i = 0; i <= this.model.getSize().width; i++)
318          {
319             this.paints.drawPeg(g, pegArea);
320             pegArea.x += this.csm.getCellWidth();
321             // We draw more pegs than walls.
322             if (i == 0)
323                continue;
324             this.paints.drawWallSet(g, wallArea);
325             wallArea.x += this.csm.getCellWidth();
326          }
327    
328          pegArea.setLocation(0, this.csm.getCellHeight());
329          wallArea.setSize(this.csm.getWallWidth(), this.csm.getCellHeightInner());
330          wallArea.setLocation(0, this.csm.getWallHeight());
331          // Draw the left side column of pegs and walls.
332          for (int i = 1; i <= this.model.getSize().height; i++)
333          {
334             this.paints.drawPeg(g, pegArea);
335             pegArea.y += this.csm.getCellHeight();
336             this.paints.drawWallSet(g, wallArea);
337             wallArea.y += this.csm.getCellHeight();
338          }
339       }

341       /**
342        * @param g
343        * @param paint
344        * @param path
345        * @param offset
346        */
347       private void drawPath(final Graphics2D g, final Paint paint, final List<MazeCell> path,
348             final int offset)
349       {
350          if (path != null && !path.isEmpty())
351          {
352             g.setPaint(paint);
353    
354             MazeCell here = path.get(0);
355             MazeCell there;
356             int x, y;
357             int width, height;
358             for (int i = 1; i < path.size(); i++)
359             {
360                there = path.get(i);
361                final Point center = this.getCellCenterInner(here);
362                if (here.getX() < there.getX())
363                {
364                   //here is west of there, going east.
365                   x = center.x + this.csm.getWallWidthHalf();
366                   y = center.y - this.csm.getWallHeightHalf();
367                   width = this.csm.getCellWidth();
368                   height = this.csm.getWallHeight();
369                }
370                else if (here.getX() > there.getX())
371                {
372                   //here is east of there, going west.
373                   x = center.x - this.csm.getWallWidthHalf() - this.csm.getCellWidth();
374                   y = center.y - this.csm.getWallHeightHalf();
375                   width = this.csm.getCellWidth();
376                   height = this.csm.getWallHeight();
377                }
378                else if (here.getY() > there.getY())
379                {
380                   //here is south of there, going north.
381                   x = center.x - this.csm.getWallWidthHalf();
382                   y = center.y - this.csm.getWallHeightHalf() - this.csm.getCellHeight();
383                   width = this.csm.getWallWidth();
384                   height = this.csm.getCellHeight();
385                }
386                else
387                {
388                   //here is north of there, going south.
389                   x = center.x - this.csm.getWallWidthHalf();
390                   y = center.y + this.csm.getWallHeightHalf();
391                   width = this.csm.getWallWidth();
392                   height = this.csm.getCellHeight();
393                }
394                g.fillRect(x - offset, y + offset, width, height);
395                here = there;
396             }
397          }
398       }

400       /**
401        * Draws the very top of the maze view. This is the most frequently called
402        * draw method and is called every time the view is repainted. Because of
403        * this it should run as fast as possible. It is much better to draw in the
404        * <code>drawCell()</code> method.
405        * @param g What to draw on.
406        */
407       private void drawTopLayer(final Graphics2D g)
408       {
409          if (this.drawPathFirst && this.robotPathModel != null)
410          {
411             this.drawPath(g,
412                           this.paints.getRunFirst(),
413                           this.robotPathModel.getPathFirst(),
414                           this.csm.getWallWidth());
415          }
416          if (this.drawPathBest && this.robotPathModel != null)
417          {
418             this.drawPath(g,
419                           this.paints.getRunBest(),
420                           this.robotPathModel.getPathBest(),
421                           -this.csm.getWallWidth());
422          }
423          if (this.drawPathCurrent && this.robotPathModel != null)
424          {
425             this.drawPath(g, this.paints.getRunCurrent(), this.robotPathModel.getPathRecent(), 0);
426          }
427          if (this.drawUnderstanding)
428          {
429             this.drawUnderstanding(g);
430          }
431          if (this.getRobotLocation() != null)
432          {
433             this.paints.drawRobot(g, this.getRobotLocation(), this.getRobotRotation());
434          }
435       }

437       /**
438        * Draws the arrows and numbers on the maze.
439        * @param g
440        */
441       private void drawUnderstanding(final Graphics2D g)
442       {
443          MazeCell here;
444          if (understandingInt != null)
445          {
446             int local;
447             for (int i = 1; i <= model.getSize().width; i++)
448             {
449                for (int j = 1; j <= model.getSize().height; j++)
450                {
451                   here = new MazeCell(i, j);
452                   g.setColor(Color.BLACK);
453                   final Point center = this.getCellCenterInner(here);
454                   local = understandingInt[i - 1][j - 1];
455                   g.drawString(String.valueOf(local), center.x - 6, center.y + 2);
456                }
457             }
458          }
459          else if (understandingDir != null)
460          {
461             Direction local;
462             for (int i = 1; i <= model.getSize().width; i++)
463             {
464                for (int j = 1; j <= model.getSize().height; j++)
465                {
466                   here = new MazeCell(i, j);
467                   final Point center = this.getCellCenterInner(here);
468                   g.setColor(Color.BLACK);
469                   local = understandingDir[i - 1][j - 1];
470                   if (local != null)
471                   {
472                      drawArrow(g, local, center.x, center.y);
473                   }
474                }
475             }
476          }
477       }

488       /**
489        * Gets all the cells that are adjacent to the given one.
490        * @param cell
491        * @return
492        */
493       @SuppressWarnings("unused")
494       private MazeCell[] getAdjacentCells(final MazeCell cell)
495       {
496          final Direction[] dirs = this.getAdjacentDirections(cell).toArray(new Direction[0]);
497          final MazeCell[] result = new MazeCell[dirs.length];
498          for (int i = 0; i < dirs.length; i++)
499          {
500             result[i] = cell.neighbor(dirs[i]);
501          }
502          return result;
503       }

505       /**
506        * Gets a direction for each adjacent cell that exists. The only time one
507        * doesn't exists is when the given cell is on the maze edge.
508        * @param cell
509        * @return
510        */
511       private EnumSet<Direction> getAdjacentDirections(final MazeCell cell)
512       {
513          // We start with all and then remove because it is less likely.
514          final EnumSet<Direction> directions = EnumSet.allOf(Direction.class);
515          if (cell.getX() == 1)
516             directions.remove(Direction.West);
517          if (cell.getY() == 1)
518             directions.remove(Direction.North);
519          if (cell.getX() == this.model.getSize().width)
520             directions.remove(Direction.East);
521          if (cell.getY() == this.model.getSize().height)
522             directions.remove(Direction.South);
523    
524          return directions;
525       }

527       /**
528        * Get the graphics object that draws onto the background image. All drawing
529        * onto the background image must use this.
530        * @return
531        */
532       private Graphics2D getBackgroundGraphics()
533       {
534          if (this.backgroundImage == null)
535          {
536             // If creating a new background image make sure we paint it.
537             this.repaintAll = true;
538             this.backgroundGraphics = null;
539             this.backgroundImage = new BufferedImage(getWidth(),
540                                                      getHeight(),
541                                                      BufferedImage.TYPE_INT_ARGB);
542          }
543          if (this.backgroundGraphics == null)
544          {
545             this.backgroundGraphics = this.backgroundImage.createGraphics();
546             this.backgroundGraphics.setRenderingHint(RenderingHints.KEY_RENDERING,
547                                                      RenderingHints.VALUE_RENDER_QUALITY);
548             this.backgroundGraphics.setRenderingHint(RenderingHints.KEY_ANTIALIASING,
549                                                      RenderingHints.VALUE_ANTIALIAS_ON);
550          }
551          return this.backgroundGraphics;
552       }

649       /**
650        * @param cell
651        * @param wall
652        * @return
653        */
654       private Rectangle getWallArea(final MazeCell cell, final Direction wall)
655       {
656          switch (wall)
657          {
658             case East :
659                return new Rectangle(cell.getX() * this.csm.getCellWidth(),
660                                     this.csm.getWallHeight() +
661                                           cell.getYZeroBased() *
662                                           this.csm.getCellHeight(),
663                                     this.csm.getWallWidth(),
664                                     this.csm.getCellHeightInner());
665             case South :
666                return new Rectangle(this.csm.getWallWidth() +
667                                           cell.getXZeroBased() *
668                                           this.csm.getCellWidth(),
669                                     cell.getY() * this.csm.getCellHeight(),
670                                     this.csm.getCellWidthInner(),
671                                     this.csm.getWallHeight());
672             default :
673                throw new IllegalArgumentException("Non supported direction: " + wall);
674          }
675       }

686       /**
687        * Invalidate a cell and mark it to be redrawn.
688        * @param cell The cell to be redrawn.
689        */
690       protected void invalidateCell(final MazeCell cell)
691       {
692          if (cell != null && this.model != null && cell.isInRange(this.model.getSize()))
693          {
694             synchronized (this.invalidatedCells)
695             {
696                this.invalidatedCells.add(cell);
697             }
698             super.repaint();
699          }
700       }

712       /**
713        * Master starting point for the custom painting. We want to draw as little
714        * as we can to keep performance up.
715        */
716       @Override
717       protected void paintComponent(final Graphics arg)
718       {
719          final Graphics2D g = (Graphics2D) arg;
720          final Graphics2D bgg = this.getBackgroundGraphics();
721          if (this.repaintAll)
722          {
723             this.repaintAll = false;
724             this.drawOutsideWalls(bgg);
725             this.drawAllCells(bgg);
726             // Because the draw all can take so long we still want to redraw invalidated cells after.
727          }
728          this.drawInvalidatedCells(bgg);
729          g.drawImage(this.backgroundImage, null, 0, 0);
730          this.drawTopLayer(g);
731       }
