126             this.csm.setCellWidth( (this.getWidth() - this.csm.getWallWidth()) /
127                                   this.model.getSize().width);
128             this.csm.setCellHeight( (this.getHeight() - this.csm.getWallHeight()) /
129                                    this.model.getSize().height);

131             final int wallSize = Math.min(csm.getCellWidth(), csm.getCellHeight()) / WALL_SIZE_DIVIDER;

133             this.csm.setWallHeight(wallSize);
134             this.painter.setMazeSize(this.getMazeSize());
135             this.repaint();
136          }

143       /**
144        * Calls the draw cell method on every cell in the maze.
145        * @param g What do draw on.
146        */
147       private void drawAllCells(final Graphics2D g)
148       {
149          //Loop through each cell in the maze.
150          for (int x = 1; x <= this.model.getSize().width; x++)
151          {
152             for (int y = 1; y <= this.model.getSize().height; y++)
153             {
154                final MazeCell cell = new MazeCell(x, y);
155                this.drawCell(g, cell);
156             } //End y loop.
157          } //End x loop.
158       }

160       /**
161        * Draws an arrow graphic.
162        * @param g What do draw on.
163        * @param local Direction to point the arrow.
164        * @param x Horizontal pixel location to draw arrow.
165        * @param y Vertical pixel location to draw arrow.
166        */
167       private void drawArrow(final Graphics2D g, final Direction local, final int x, final int y)
168       {
169          //Draws an arrow in the direction of "local" centered on the point (x,y)
170          if (local.equals(Direction.North))
171          {
172             final int[] ys =
173             {
174                y + this.csm.getCellHeight() * 3 / 8, y, y, y - this.csm.getCellHeight() * 3 / 8, y, y
175             };
176             final int[] xs =
177             {
178                x, x - this.csm.getCellWidth() / 8, x - this.csm.getCellWidth() / 4, x,
179                x + this.csm.getCellWidth() / 4, x + this.csm.getCellWidth() / 8
180             };
181             g.drawPolygon(xs, ys, 6);
182          }
183          if (local.equals(Direction.South))
184          {
185             final int[] ys =
186             {
187                y - this.csm.getCellHeight() * 3 / 8, y, y, y + this.csm.getCellHeight() * 3 / 8, y, y
188             };
189             final int[] xs =
190             {
191                x, x - this.csm.getCellWidth() / 8, x - this.csm.getCellWidth() / 4, x,
192                x + this.csm.getCellWidth() / 4, x + this.csm.getCellWidth() / 8
193             };
194             g.drawPolygon(xs, ys, 6);
195          }
196          if (local.equals(Direction.West))
197          {
198             final int[] xs =
199             {
200                x + this.csm.getCellWidth() * 3 / 8, x, x, x - this.csm.getCellWidth() * 3 / 8, x, x
201             };
202             final int[] ys =
203             {
204                y, y - this.csm.getCellHeight() / 8, y - this.csm.getCellHeight() / 4, y,
205                y + this.csm.getCellHeight() / 4, y + this.csm.getCellHeight() / 8
206             };
207             g.drawPolygon(xs, ys, 6);
208          }
209          if (local.equals(Direction.East))
210          {
211             final int[] xs =
212             {
213                x - this.csm.getCellWidth() * 3 / 8, x, x, x + this.csm.getCellWidth() * 3 / 8, x, x
214             };
215             final int[] ys =
216             {
217                y, y - this.csm.getCellHeight() / 8, y - this.csm.getCellHeight() / 4, y,
218                y + this.csm.getCellHeight() / 4, y + this.csm.getCellHeight() / 8
219             };
220             g.drawPolygon(xs, ys, 6);
221          }
222       }

224       /**
225        * The primary draw method for a cell. This should draw all aspects of a
226        * cell.
227        * @param g Where to draw.
228        * @param cell The cell in question.
229        */
230       private void drawCell(final Graphics2D g, final MazeCell cell)
231       {
232          this.painter.drawCellBackground(g, this.getCellAreaInner(cell));
233    
234          if (this.model.getWall(cell, Direction.East).isSet())
235          {
236             this.painter.drawWallSet(g, this.getWallArea(cell, Direction.East));
237          }
238          else
239          {
240             this.painter.drawWallEmpty(g, this.getWallArea(cell, Direction.East));
241          }
242          if (this.model.getWall(cell, Direction.South).isSet())
243          {
244             this.painter.drawWallSet(g, this.getWallArea(cell, Direction.South));
245          }
246          else
247          {
248             this.painter.drawWallEmpty(g, this.getWallArea(cell, Direction.South));
249          }
250          this.painter.drawPeg(g, this.getPegArea(cell));
251          if (this.robotPathModel != null)
252          {
253             //Draw the fog of war.
254             if (this.drawFog && !this.robotPathModel.hasCellBeenVisited(cell))
255             {
256                final Rectangle area = this.getCellArea(cell);
257                final MazeCell east = cell.neighbor(Direction.East);
258                final MazeCell south = cell.neighbor(Direction.South);
259                if (east.isInRange(this.model.getSize()) &&
260                    this.robotPathModel.hasCellBeenVisited(east))
261                {
262                   area.width -= this.csm.getWallWidth();
263                }
264                if (south.isInRange(this.model.getSize()) &&
265                    this.robotPathModel.hasCellBeenVisited(south))
266                {
267                   area.height -= this.csm.getWallHeight();
268                }
269                this.painter.drawFog(g, area);
270             }
271             // Draw a current path of dots.
272             if (this.drawPathCurrent && this.robotPathModel.getPathRecent().contains(cell))
273             {
274                final EnumSet<Direction> directions = EnumSet.noneOf(Direction.class);
275                for (final Direction dir : this.getAdjacentDirections(cell))
276                {
277                   if (!this.model.getWall(cell, dir).isSet() &&
278                       this.robotPathModel.getPathRecent().contains(cell.neighbor(dir)))
279                   {
280                      directions.add(dir);
281                   }
282                }
283                //this.paints.drawRunCurrent(g, this.getCellAreaInner(cell), directions);
284             }
285          }
286       }

288       /**
289        * Draws all cells that have been invalidated.
290        * @param g Where to draw.
291        */
292       private void drawInvalidatedCells(final Graphics2D g)
293       {
294          synchronized (this.invalidatedCells)
295          {
296             for (final MazeCell cell : this.invalidatedCells)
297             {
298                this.drawCell(g, cell);
299                //System.out.println("Drew cell: " + cell);
300             }
301             this.invalidatedCells.clear();
302          }
303       }

305       /**
306        * Draws the top and left outside walls as these don't fall inside of any
307        * cells.
308        * @param g Where to draw.
309        */
310       private void drawOutsideWalls(final Graphics2D g)
311       {
312    
313          final Rectangle pegArea = new Rectangle(0,
314                                                  0,
315                                                  this.csm.getWallWidth(),
316                                                  this.csm.getWallHeight());
317          final Rectangle wallArea = new Rectangle(this.csm.getWallWidth(),
318                                                   0,
319                                                   this.csm.getCellWidthInner(),
320                                                   this.csm.getWallHeight());
321    
322          for (int i = 0; i <= this.model.getSize().width; i++)
323          {
324             this.painter.drawPeg(g, pegArea);
325             pegArea.x += this.csm.getCellWidth();
326             // We draw more pegs than walls.
327             if (i == 0)
328             {
329                continue;
330             }
331             this.painter.drawWallSet(g, wallArea);
332             wallArea.x += this.csm.getCellWidth();
333          }
334    
335          pegArea.setLocation(0, this.csm.getCellHeight());
336          wallArea.setSize(this.csm.getWallWidth(), this.csm.getCellHeightInner());
337          wallArea.setLocation(0, this.csm.getWallHeight());
338          // Draw the left side column of pegs and walls.
339          for (int i = 1; i <= this.model.getSize().height; i++)
340          {
341             this.painter.drawPeg(g, pegArea);
342             pegArea.y += this.csm.getCellHeight();
343             this.painter.drawWallSet(g, wallArea);
344             wallArea.y += this.csm.getCellHeight();
345          }
346       }

348       /**
349        * Get the cell that the given point is located in. This gives us a means to
350        * translate between coordinate positions and cells.
351        * @param location A coordinate point in the maze area.
352        * @return The cell or null if the point is not inside a valid cell.
353        */
354       @SuppressWarnings("unused")
355       private MazeCell getHostCell(Point location)
356       {
357          try
358          {
359             MazeCell cell = new MazeCell( ( (location.x - csm.getWallWidth()) / csm.getCellWidth()) + 1,
360                                          ( (location.y - csm.getWallHeight()) / csm.getCellHeight()) + 1);
361             if (cell.isInRange(model.getSize()))
362             {
363                return cell;
364             }
365          }
366          catch (Exception e)
367          {}
368          return null;
369       }

371       /**
372        * Draw a robot path onto the maze. The path will have the width of the cell
373        * walls.
374        * @param g Where to draw.
375        * @param path The cell to cell path to draw.
376        * @param offset Shift the path by the given amount so that you can draw
377        *           multiple paths without them overlapping. A value of 0 draws to
378        *           the center of the cell.
379        * @param trimTail If set to true the tail of the path will be trimmed to the
380        *           location of the robot. This allows the current path to finish
381        *           right on the current robot position.
382        */
383       private void drawPath(final Graphics2D g, final List<MazeCell> path, final int offset,
384             boolean trimTail)
385       {
386          if (path != null && !path.isEmpty())
387          {
388             MazeCell here = path.get(0);
389             MazeCell there;
390             int x, y;
391             int width, height;
392             for (int i = 1; i < path.size(); i++)
393             {
394                there = path.get(i);
395                final Point center = this.getCellCenterInner(here);
396                if (here.getX() < there.getX())
397                {
398                   //here is west of there, going east.
399                   x = center.x + this.csm.getWallWidthHalf();
400                   y = center.y - this.csm.getWallHeightHalf();
401                   width = this.csm.getCellWidth();
402                   height = this.csm.getWallHeight();
403                }
404                else if (here.getX() > there.getX())
405                {
406                   //here is east of there, going west.
407                   x = center.x - this.csm.getWallWidthHalf() - this.csm.getCellWidth();
408                   y = center.y - this.csm.getWallHeightHalf();
409                   width = this.csm.getCellWidth();
410                   height = this.csm.getWallHeight();
411                }
412                else if (here.getY() > there.getY())
413                {
414                   //here is south of there, going north.
415                   x = center.x - this.csm.getWallWidthHalf();
416                   y = center.y - this.csm.getWallHeightHalf() - this.csm.getCellHeight();
417                   width = this.csm.getWallWidth();
418                   height = this.csm.getCellHeight();
419                }
420                else
421                {
422                   //here is north of there, going south.
423                   x = center.x - this.csm.getWallWidthHalf();
424                   y = center.y + this.csm.getWallHeightHalf();
425                   width = this.csm.getWallWidth();
426                   height = this.csm.getCellHeight();
427                }
428                // If we are at the last cell and we are trimming the tail up to the robot.
429                if (i == path.size() - 1 && trimTail)
430                {
431                   if (here.getX() < there.getX())
432                   {
433                      //here is west of there, going east.
434                      width = Math.abs(x - this.robotLocation.x);
435                   }
436                   else if (here.getX() > there.getX())
437                   {
438                      //here is east of there, going west.
439                      width -= this.robotLocation.x - x;
440                      x = this.robotLocation.x;
441    
442                   }
443                   else if (here.getY() > there.getY())
444                   {
445                      //here is south of there, going north.
446                      height -= this.robotLocation.y - y;
447                      y = this.robotLocation.y;
448                   }
449                   else
450                   {
451                      //here is north of there, going south.
452                      height = this.robotLocation.y - y;
453                   }
454                }
455                g.fillRect(x - offset, y + offset, width, height);
456                here = there;
457             }
458          }
459       }

461       /**
462        * Draws the very top of the maze view. This is the most frequently called
463        * draw method and is called every time the view is repainted. Because of
464        * this it should run as fast as possible. It is much better to draw in the
465        * <code>drawCell()</code> method.
466        * @param g What to draw on.
467        */
468       private void drawTopLayer(final Graphics2D g)
469       {
470          this.setRenderingQualityLow(g);
471          if (this.robotPathModel != null)
472          {
473             if (this.drawPathFirst)
474             {
475                g.setPaint(this.painter.getRunFirst());
476                this.drawPath(g, this.robotPathModel.getPathFirst(), this.csm.getWallWidth(), false);
477             }
478             if (this.drawPathBest)
479             {
480                g.setPaint(this.painter.getRunBest());
481                this.drawPath(g, this.robotPathModel.getPathBest(), -this.csm.getWallWidth(), false);
482             }
483             if (this.drawPathCurrent)
484             {
485                g.setPaint(this.painter.getRunCurrent());
486                this.drawPath(g, this.robotPathModel.getPathRecent(), 0, true);
487             }
488          }
489          if (this.drawUnderstanding)
490          {
491             this.drawUnderstanding(g);
492          }
493          if (this.getRobotLocation() != null)
494          {
495             this.setRenderingQualityHigh(g);
496             this.painter.drawRobot(g,
497                                    this.getRobotLocation(),
498                                    this.getRobotRotation(),
499                                    this.csm.getCellWidthInner(),
500                                    this.csm.getCellHeightInner());
501          }
502       }

524       /**
525        * Draws the arrows and numbers on the maze.
526        * @param g What to draw on.
527        */
528       private void drawUnderstanding(final Graphics2D g)
529       {
530          MazeCell here;
531          if (understandingInt != null)
532          {
533             int local;
534             for (int i = 1; i <= model.getSize().width; i++)
535             {
536                for (int j = 1; j <= model.getSize().height; j++)
537                {
538                   here = new MazeCell(i, j);
539                   g.setColor(Color.BLACK);
540                   final Point center = this.getCellCenterInner(here);
541                   local = understandingInt[i - 1][j - 1];
542                   g.drawString(String.valueOf(local), center.x - 6, center.y + 2);
543                }
544             }
545          }
546          else if (understandingDir != null)
547          {
548             Direction local;
549             for (int i = 1; i <= model.getSize().width; i++)
550             {
551                for (int j = 1; j <= model.getSize().height; j++)
552                {
553                   here = new MazeCell(i, j);
554                   final Point center = this.getCellCenterInner(here);
555                   g.setColor(Color.BLACK);
556                   local = understandingDir[i - 1][j - 1];
557                   if (local != null)
558                   {
559                      drawArrow(g, local, center.x, center.y);
560                   }
561                }
562             }
563          }
564       }

575       /**
576        * Gets all the cells that are adjacent to the given one.
577        * @param cell The cell in question.
578        * @return All adjacent or connecting cells.
579        */
580       @SuppressWarnings("unused")
581       private MazeCell[] getAdjacentCells(final MazeCell cell)
582       {
583          final Direction[] dirs = this.getAdjacentDirections(cell).toArray(new Direction[0]);
584          final MazeCell[] result = new MazeCell[dirs.length];
585          for (int i = 0; i < dirs.length; i++)
586          {
587             result[i] = cell.neighbor(dirs[i]);
588          }
589          return result;
590       }

592       /**
593        * Gets a direction for each adjacent cell that exists. The only time one
594        * doesn't exists is when the given cell is on the maze edge.
595        * @param cell The cell in question.
596        * @return A set of directions with each direction confirming the existence
597        *         of a neighbor cell.
598        */
599       private EnumSet<Direction> getAdjacentDirections(final MazeCell cell)
600       {
601          // We start with all and then remove because it is less likely.
602          final EnumSet<Direction> directions = EnumSet.allOf(Direction.class);
603          if (cell.getX() == 1)
604          {
605             directions.remove(Direction.West);
606    
607          }
608          if (cell.getY() == 1)
609          {
610             directions.remove(Direction.North);
611    
612          }
613          if (cell.getX() == this.model.getSize().width)
614          {
615             directions.remove(Direction.East);
616    
617          }
618          if (cell.getY() == this.model.getSize().height)
619          {
620             directions.remove(Direction.South);
621    
622          }
623          return directions;
624       }

626       /**
627        * Get the graphics object that draws onto the background image. All drawing
628        * onto the background image must use this.
629        * @return The graphics object.
630        */
631       private Graphics2D getBackgroundGraphics()
632       {
633          if (this.backgroundImage == null)
634          {
635             // If creating a new background image make sure we paint it.
636             this.repaintAll = true;
637             this.backgroundGraphics = null;
638             this.backgroundImage = new BufferedImage(getWidth(),
639                                                      getHeight(),
640                                                      BufferedImage.TYPE_INT_ARGB);
641          }
642          if (this.backgroundGraphics == null)
643          {
644             this.backgroundGraphics = this.backgroundImage.createGraphics();
645             this.setRenderingQualityHigh(this.backgroundGraphics);
646          }
647          return this.backgroundGraphics;
648       }

653        * @return The location and area in pixels of where the cell is located.

655       private Rectangle getCellArea(final MazeCell cell)

659                                     cell.getYZeroBased() *
660                                     this.csm.getCellHeight(),
661                               this.csm.getCellWidth(),
662                               this.csm.getCellHeight());
663       }

665       /**
666        * Similar to <code>getCellArea</code> but just the inside of the cell without the
667        * walls.
668        * @param cell The cell in question.
669        * @return Cell area minus the walls and corner peg.
670        */
671       private Rectangle getCellAreaInner(final MazeCell cell)
672       {
673          return new Rectangle(this.csm.getWallWidth() + cell.getXZeroBased() * this.csm.getCellWidth(),
674                               this.csm.getWallHeight() +
675                                     cell.getYZeroBased() *

749       /**
750        * Get the area of a cell wall.
751        * @param cell The cell in question.
752        * @param wall Which wall do you want. Must be East or South.
753        * @return The absolute coordinates of the area.
754        */
755       private Rectangle getWallArea(final MazeCell cell, final Direction wall)
756       {
757          switch (wall)
758          {
759             case East :
760                return new Rectangle(cell.getX() * this.csm.getCellWidth(),
761                                     this.csm.getWallHeight() +
762                                           cell.getYZeroBased() *
763                                           this.csm.getCellHeight(),
764                                     this.csm.getWallWidth(),
765                                     this.csm.getCellHeightInner());
766             case South :
767                return new Rectangle(this.csm.getWallWidth() +
768                                           cell.getXZeroBased() *
769                                           this.csm.getCellWidth(),
770                                     cell.getY() * this.csm.getCellHeight(),
771                                     this.csm.getCellWidthInner(),
772                                     this.csm.getWallHeight());
773             default :
774                throw new IllegalArgumentException("Non supported direction: " + wall);
775          }
776       }

787       /**
788        * Invalidate a cell and mark it to be redrawn.
789        * @param cell The cell to be redrawn.
790        */
791       protected void invalidateCell(final MazeCell cell)
792       {
793          if (cell != null && this.model != null && cell.isInRange(this.model.getSize()))
794          {
795             synchronized (this.invalidatedCells)
796             {
797                this.invalidatedCells.add(cell);
798             }
799             super.repaint();
800          }
801       }

813       /**
814        * Master starting point for the custom painting. We want to draw as little
815        * as we can to keep performance up.
816        */
817       @Override
818       protected void paintComponent(final Graphics arg)
819       {
820          final Graphics2D bgg = this.getBackgroundGraphics();
821          if (this.repaintAll)
822          {
823             this.repaintAll = false;
824             this.drawOutsideWalls(bgg);
825             this.drawAllCells(bgg);
826             // Because the draw all can take so long we still want to redraw invalidated cells after.
827          }
828          this.drawInvalidatedCells(bgg);
829          final Graphics2D g = (Graphics2D) arg;
830          g.drawImage(this.backgroundImage, null, 0, 0);
831          this.drawTopLayer(g);
832       }
