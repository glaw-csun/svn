001    package maze.gui;
002    
003    import java.io.File;
004    
005    /**
006     * Provides a common set of methods for the maze editor and code editor panels
007     * that provide access to file operations from the main file menu.
008     * @author Johnathan Smith
009     */
010    public interface MenuControlled
011    {
012       /**
013        * Closes the currently displayed item.
014        */
015       public void close();
016    
017       /**
018        * Get a description for the types of files that this panel supports.
019        * @return Description.
020        */
021       public String getFileTypeDescription();
022    
023       /**
024        * Test whether the given file is of the type that this page supports.
025        * @param file File to test.
026        * @return true if it is a recognized file type.
027        */
028       public boolean isMyFileType(File file);
029    
030       /**
031        * Open the file.
032        * @param file The file to open.
033        */
034       public void open(File file);
035    
036       /**
037        * Save the currently displayed item.
038        */
039       public void saveCurrent();
040    
041       /**
042        * Perform cleanup operations like saving files before the application exits.
043        * @return true if this page is ready for application exit, false if the exit
044        *         should be stopped.
045        */
046       public boolean canExit();
047    }
