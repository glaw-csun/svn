001    package maze.gui;
002    
003    import java.awt.Color;
004    import java.awt.Dimension;
005    import java.awt.GradientPaint;
006    import java.awt.Point;
007    
008    import maze.Main;
009    import maze.model.CellSizeModelInterface;
010    
011    public class MazePainterDefault extends MazePainter
012    {
013    
014       public MazePainterDefault(CellSizeModelInterface cellSizeModel)
015       {
016          super(cellSizeModel);
017       }
018    
019       /**
020        * Set the default values for colors. You may also want to call
021        * <code>setGradients(...)</code> to completely initialize all paints.
022        */
023       public void setDefaults()
024       {
025          this.cellBackground = Color.BLACK;
026          this.wallSet = Color.BLACK;
027          this.wallEmpty = Color.BLACK;
028          this.background = Color.WHITE;
029          this.peg = Color.BLACK;
030          this.pegValid = Color.BLACK;
031          this.pegInvalid = new GradientPaint(0, 0, Color.RED, 2, 2, new Color(240, 0, 0), true);
032          this.fog = new Color(0, 0, 0, 75);
033          this.runFirst = new Color(0, 0, 255, 75);
034          this.runBest = new Color(255, 0, 0, 75);
035          this.runCurrent = new Color(0, 255, 0, 50);
036          this.hover = new Color(255, 255, 0, 100);
037          this.robotImage = Main.getImageResource("gui/images/mouse.png");
038       }
039    
040       /**
041        * When the maze size changes we want to recalculate our gradients.
042        * {@inheritDoc}
043        */
044       public void setMazeSize(Dimension size)
045       {
046          if (this.mazeWidth != size.width || this.mazeHeight != size.height)
047          {
048             this.mazeWidth = size.width;
049             this.mazeHeight = size.height;
050             final int halfX = this.mazeWidth / 2;
051             final int halfY = this.mazeHeight / 2;
052             final Point zero = new Point(0, 0);
053             final Point middle = new Point(halfX, halfY);
054             this.cellBackground = new GradientPaint(zero,
055                                                     Color.WHITE,
056                                                     middle,
057                                                     new Color(229, 236, 255),
058                                                     true);
059             this.wallEmpty = new GradientPaint(zero, //Gradient start corner.
060                                                new Color(229, 236, 255), //Really light blue.
061                                                middle,
062                                                new Color(204, 218, 255), //Light blue.
063                                                true);
064             this.wallSet = new GradientPaint(zero,
065                                              new Color(0, 94, 189),
066                                              middle,
067                                              new Color(0, 56, 112),
068                                              true);
069          }
070       }
071    
072    }
