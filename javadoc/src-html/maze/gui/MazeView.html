177       /**
178        * Draws the entire maze onto the given graphics device.
179        */
180       private void redrawAll(final Graphics2D g)
181       {
182          final int mazeWidth = this.model.getSize().width * this.csm.getCellWidth();
183          final int mazeHeight = this.model.getSize().height * this.csm.getCellHeight();
184          final Dimension modelSize = this.model.getSize();
185    
186          //Hopefully it won't hurt to always have this on.
187          g.setComposite(AlphaComposite.SrcOver);
188    
189          //Enabling this line will turn off transparency which will look ugly but allows a performance comparison.
190          //g.setComposite(AlphaComposite.Src);
191    
192          this.paints.setMazeSize(this.getMazeSize());
193    
194          if (this.background == null)
195          {
196             this.background = new BufferedImage(getWidth(), getHeight(), BufferedImage.TYPE_INT_ARGB);
197             Graphics2D bgg = (Graphics2D) this.background.getGraphics();
198             // Fill in the background color.
199             bgg.setColor(Color.white);
200             bgg.fillRect(0, 0, super.getWidth(), super.getHeight());
201             //Paint the gradient maze background.
202             bgg.setPaint(this.paints.getCellBackground());
203             bgg.fillRect(0, 0, mazeWidth, mazeHeight);
204    
205             // Draw inactive walls
206             bgg.setPaint(this.paints.getWallEmpty());
207             for (int i = 0; i < modelSize.width + 1; i++)
208             {
209                //Draw vertical walls.
210                bgg.fillRect(i * this.csm.getCellWidth() - this.csm.getWallWidthHalf(),
211                             0,
212                             this.csm.getWallWidth(),
213                             mazeHeight);
214             }
215             for (int i = 0; i < modelSize.height + 1; i++)
216             {
217                //Draw horizontal walls.
218                bgg.fillRect(0,
219                             i * this.csm.getCellHeight() - this.csm.getWallHeightHalf(),
220                             mazeWidth,
221                             this.csm.getWallHeight());
222             }
223    
224             // Draw outer walls
225             bgg.setPaint(this.paints.getWallSet());
226             bgg.fillRect(-csm.getWallWidthHalf(), -csm.getWallHeightHalf(), mazeWidth, csm.getWallHeight());
227             bgg.fillRect(-csm.getWallWidthHalf(), -csm.getWallHeightHalf(), csm.getWallWidth(), mazeHeight);
228             bgg.fillRect(0, mazeHeight - csm.getWallHeightHalf(), mazeWidth, csm.getWallHeight());
229             bgg.fillRect(mazeWidth - csm.getWallWidthHalf(), 0, csm.getWallWidth(), mazeHeight);
230    
231             // Draw pegs
232             bgg.setPaint(this.paints.getPeg());
233             for (int i = 1; i <= modelSize.width + 1; i++)
234                for (int j = 1; j <= modelSize.height + 1; j++)
235                   bgg.fill(this.getPegRegion(new MazeCell(i, j), PegLocation.TopLeft));
236          } //End update background image.
237    
238          g.drawImage(background, 0, 0, null);
239    
240          //Loop through each cell in the maze.
241          for (int x = 1; x <= modelSize.width; x++)
242          {
243             for (int y = 1; y <= modelSize.height; y++)
244             {
245                final MazeCell cell = new MazeCell(x, y);
246    
247                g.setPaint(this.paints.getWallSet());
248                if (this.model.getWall(cell, Direction.South).isSet())
249                   g.fill(this.getWallLocation(cell, Direction.South));
250                if (this.model.getWall(cell, Direction.East).isSet())
251                   g.fill(this.getWallLocation(cell, Direction.East));
252    
253             } //End y loop.
254          } //End x loop.
255    
256          //Draw the yellow hover box. This could get axed.
257          if (this.active != null)
258          {
259             g.setPaint(this.paints.getHover());
260             g.fillRect(active.getXZeroBased() * this.csm.getCellWidth() + this.csm.getWallWidthHalf(),
261                        active.getYZeroBased() * this.csm.getCellHeight() + this.csm.getWallHeightHalf(),
262                        this.csm.getCellWidth() - this.csm.getWallWidth(),
263                        this.csm.getCellHeight() - this.csm.getWallHeight());
264          }
265    
266          if (drawFog == true)
267          {
268             drawFog(g);
269          }
270    
271          if (drawFirstRun == true)
272          {
273             drawFirstRun(g);
274          }
275    
276          if (drawBestRun == true)
277          {
278             drawBestRun(g);
279          }
280    
281          if (drawCurrentRun == true)
282          {
283             drawCurrentRun(g);
284          }
285    
286          if (drawUnderstanding == true)
287          {
288             drawUnderstanding(g);
289          }
290    
291          // Draw the robot onto the maze.
292          if (this.editable == false && this.robotLocation != null)
293          {
294             //We have to translate our coordinate system to make the image rotation easier.
295             g.translate(this.robotLocation.x, this.robotLocation.y);
296    
297             //Back up the current graphics state.
298             final AffineTransform oldTransform = g.getTransform();
299             //We add half a PI to the rotation because the top of the image is forward.
300             final AffineTransform transform = AffineTransform.getRotateInstance(this.robotRotation +
301                                                                                 Math.PI /
302                                                                                 2);
303             //Set the image rotation transformation.
304             g.transform(transform);
305             //Get the size of half the final image. Based on smallest cell dimension.
306             final int size = Math.min(this.csm.getCellWidth() - this.csm.getWallWidth(),
307                                       this.csm.getCellHeight() - this.csm.getWallHeight()) / 2;
308             //Draw the image to scale. The point 0,0 is the center of the image.
309             g.drawImage(this.robotImage.getImage(),
310                         -size,
311                         -size,
312                         size,
313                         size,
314                         0,
315                         0,
316                         this.robotImage.getIconWidth(),
317                         this.robotImage.getIconHeight(),
318                         null);
319             //Restore the original graphics state.
320             g.setTransform(oldTransform);
321             g.translate(-this.robotLocation.x, -this.robotLocation.y);
322          }
323    
324       } //End method.

327        * Turns a maze cell into global coordinates for the center of the cell.
328        */
329       protected Point getCellCenter(MazeCell cell)
330       {
331          return new Point( (cell.getXZeroBased() * this.csm.getCellWidth()) + this.csm.getCellWidthHalf(),
332                           (cell.getYZeroBased() * this.csm.getCellHeight()) + this.csm.getCellHeightHalf());
333       }

335       /**
336        * Get a rectangle covering the wall segment with component relative
337        * coordinates.
338        */
339       private Rectangle getWallLocation(MazeCell cell, Direction wall)
340       {
341          Point center = this.getCellCenter(cell);
342          if (wall == Direction.North)
343          {
344             return new Rectangle(center.x - (this.csm.getCellWidthHalf() - this.csm.getWallWidthHalf()),
345                                  center.y - (this.csm.getCellHeightHalf() + this.csm.getWallHeightHalf()),
346                                  this.csm.getCellWidth() - this.csm.getWallWidth(),
347                                  this.csm.getWallHeight());
348          }
349          else if (wall == Direction.South)
350          {
351             return new Rectangle(center.x - (this.csm.getCellWidthHalf() - this.csm.getWallWidthHalf()),
352                                  center.y + (this.csm.getCellHeightHalf() - this.csm.getWallHeightHalf()),
353                                  this.csm.getCellWidth() - this.csm.getWallWidth(),
354                                  this.csm.getWallHeight());
355          }
356          else if (wall == Direction.East)
357          {
358             return new Rectangle(center.x + (this.csm.getCellWidthHalf() - this.csm.getWallWidthHalf()),
359                                  center.y - (this.csm.getCellHeightHalf() - this.csm.getWallHeightHalf()),
360                                  this.csm.getWallWidth(),
361                                  this.csm.getCellHeight() - this.csm.getWallHeight());
362          }
363          else if (wall == Direction.West)
364          {
365             return new Rectangle(center.x - (this.csm.getCellWidthHalf() + this.csm.getWallWidthHalf()),
366                                  center.y - (this.csm.getCellHeightHalf() - this.csm.getWallHeightHalf()),
367                                  this.csm.getWallWidth(),
368                                  this.csm.getCellHeight() - this.csm.getWallHeight());
369          }
370          return new Rectangle(); //should never get here.
371       }

373       /**
374        * Get the absolute region of a peg with respect to the given maze cell.
375        */
376       protected Rectangle getPegRegion(MazeCell cell, PegLocation peg)
377       {
378          if (peg == PegLocation.TopLeft)
379          {
380             return new Rectangle( (cell.getX() * this.csm.getCellWidth()) -
381                                        this.csm.getWallWidthHalf() -
382                                        this.csm.getCellWidth(),
383                                  (cell.getY() * this.csm.getCellHeight()) -
384                                        this.csm.getCellHeight() -
385                                        this.csm.getWallHeightHalf(),
386                                  this.csm.getWallWidth(),
387                                  this.csm.getWallHeight());
388          }
389          else if (peg == PegLocation.TopRight)
390          {
391             return new Rectangle(cell.getX() * this.csm.getCellWidth() - this.csm.getWallWidthHalf(),
392                                  (cell.getY() * this.csm.getCellHeight()) -
393                                        this.csm.getCellHeight() -
394                                        this.csm.getWallHeightHalf(),
395                                  this.csm.getWallWidth(),
396                                  this.csm.getWallHeight());
397          }
398          else if (peg == PegLocation.BottomRight)
399          {
400             return new Rectangle(cell.getX() * this.csm.getCellWidth() - this.csm.getWallWidthHalf(),
401                                  cell.getY() * this.csm.getCellHeight() - this.csm.getWallHeightHalf(),
402                                  this.csm.getWallWidth(),
403                                  this.csm.getWallHeight());
404          }
405          else if (peg == PegLocation.BottomLeft)
406          {
407             return new Rectangle(cell.getXZeroBased() *
408                                        this.csm.getCellWidth() -
409                                        this.csm.getWallWidthHalf(),
410                                  cell.getY() * this.csm.getCellHeight() - this.csm.getWallHeightHalf(),
411                                  this.csm.getWallWidth(),
412                                  this.csm.getWallHeight());
413          }
414          return new Rectangle(); //Should never get here.
415       }

417       /**
418        * Converts a mouse pointer position into the maze cell that it is in.
419        * @throws Exception If the pointer location is not within the maze.
420        */
421       protected MazeCell getHostMazeCell(Point pointerLocation) throws Exception
422       {
423          MazeCell cell = new MazeCell( (pointerLocation.x / this.csm.getCellWidth()) + 1,
424                                       (pointerLocation.y / this.csm.getCellHeight()) + 1);
425          if (cell.isInRange(this.model.getSize()))
426          {
427             return cell;
428          }
429          else
430          {
431             throw new Exception("The pointer location is outside of the current maze.");
432          }
433       }

435       /**
436        * @param cell
437        * @param mouseLocation
438        * @return
439        * @throws java.lang.Exception If the mouse pointer isn't actually on a wall.
440        */
441       private Direction getWallDirection(MazeCell cell, Point mouseLocation) throws Exception
442       {
443          for (Direction direction : Direction.values())
444          {
445             if (getWallLocation(cell, direction).contains(mouseLocation))
446             {
447                return direction;
448             }
449          }
450          throw new Exception("Not inside a wall.");
451       }

453       /**
454        * Converts a mouse pointer location into an actual maze wall object.
455        * @param mouseLocation
456        * @return
457        * @throws java.lang.Exception If the mouse pointer isn't actually on a wall.
458        */
459       protected MazeWall getWall(Point mouseLocation) throws Exception
460       {
461          final MazeCell cell = getHostMazeCell(mouseLocation);
462          return model.getWall(cell, getWallDirection(cell, mouseLocation));
463       }

472          if (model != null)
473          {
474             background = null;
475             this.csm.setCellWidth(this.getWidth() / this.model.getSize().width);
476             this.csm.setCellHeight(this.getHeight() / this.model.getSize().height);
477

479             this.csm.setWallWidth(wallSize);
480             this.csm.setWallHeight(wallSize);
481          }
482       }

496       /**
497        * Sets whether this MazeView can modify its underlying MazeModel.
498        * @param b true if the MazeModel should be editable, false otherwise
499        */
500       public void setEditable(boolean b)
501       {
502          if (b == editable)
503             return;
504    
505          editable = b;
506          if (editable)
507          {
508             //Create an event handler to handle mouse events.
509             mouseAdapter = new MouseAdapter()
510             {
511                @Override
512                public void mouseMoved(MouseEvent e)
513                {
514                   if (model != null)
515                   {
516                      try
517                      {
518                         active = getHostMazeCell(e.getPoint());
519                         repaint();
520                      }
521                      catch (Exception ex)
522                      {}
523                   }
524                } // public void mouseMoved(MouseEvent e)
525    
526                @Override
527                public void mouseDragged(MouseEvent e)
528                {
529                   if (model != null)
530                   {
531                      try
532                      {
533                         active = getHostMazeCell(e.getPoint());
534                         MazeWall wall = getWall(e.getPoint());
535                         if (SwingUtilities.isLeftMouseButton(e))
536                            wall.set(true);
537                         else if (SwingUtilities.isRightMouseButton(e))
538                            wall.set(false);
539                         repaint();
540                      }
541                      catch (Exception ex)
542                      {}
543                   }
544                } // public void mouseDragged(MouseEvent e)
545    
546                @Override
547                public void mousePressed(MouseEvent e)
548                {
549                   if (model != null)
550                   {
551                      try
552                      {
553                         final MazeWall wall = getWall(e.getPoint());
554                         //Flip the status of the wall.
555                         wall.set(!wall.isSet());
556                         repaint();
557                      }
558                      catch (Exception ex)
559                      {}
560                   }
561                } // public void mousePressed(MouseEvent e)
562             };
563             this.addMouseListener(mouseAdapter);
564             this.addMouseMotionListener(mouseAdapter);
565          } // if (editable)
566          else
567          {
568             active = null;
569             this.removeMouseListener(mouseAdapter);
570             this.removeMouseMotionListener(mouseAdapter);
571             repaint();
572          } // else
573       }

601       private void drawFirstRun(Graphics2D g)
602       {
603          if (this.firstRun != null && this.firstRun.isEmpty() == false)
604          {
605             g.setPaint(this.paints.getRunFirst());
606    
607             MazeCell here = firstRun.get(0);
608             MazeCell there;
609             Point center;
610             int width, height;
611             for (int i = 1; i < firstRun.size(); i++)
612             {
613                there = firstRun.get(i);
614                if (here.getX() < there.getX())
615                {
616                   //here is west of there
617                   center = this.getCellCenter(here);
618                   center.x += this.csm.getCellWidth() / 4;
619                   width = this.csm.getCellWidth();
620                   height = this.csm.getCellHeight() / 4;
621                }
622                else if (here.getX() > there.getX())
623                {
624                   //here is east of there
625                   center = this.getCellCenter(there);
626                   width = this.csm.getCellWidth();
627                   height = this.csm.getCellHeight() / 4;
628                }
629                else if (here.getY() > there.getY())
630                {
631                   //here is south of there
632                   center = this.getCellCenter(there);
633                   width = this.csm.getCellWidth() / 4;
634                   height = this.csm.getCellHeight();
635                }
636                else
637                {
638                   //here is north of there
639                   center = this.getCellCenter(here);
640                   width = this.csm.getCellWidth() / 4;
641                   height = this.csm.getCellHeight();
642                }
643                g.fillRect(center.x, center.y, width, height);
644                here = there;
645             }
646          }
647       }

659       private void drawBestRun(Graphics2D g)
660       {
661          if (this.bestRun != null && this.bestRun.isEmpty() == false)
662          {
663             g.setPaint(this.paints.getRunBest());
664    
665             MazeCell here = bestRun.get(0);
666             MazeCell there;
667             int x, y;
668             int width, height;
669             for (int i = 1; i < bestRun.size(); i++)
670             {
671                there = bestRun.get(i);
672                if (here.getX() < there.getX())
673                {
674                   //here is west of there
675                   x = (here.getX() - 1) * this.csm.getCellWidth() + this.csm.getCellWidth() / 4;
676                   y = (here.getY() - 1) * this.csm.getCellHeight() + this.csm.getCellHeight() / 4;
677                   width = 5 * this.csm.getCellWidth() / 4;
678                   height = this.csm.getCellHeight() / 4;
679                }
680                else if (here.getX() > there.getX())
681                {
682                   //here is east of there
683                   x = (there.getX() - 1) * this.csm.getCellWidth() + this.csm.getCellWidth() / 4;
684                   y = (there.getY() - 1) * this.csm.getCellHeight() + this.csm.getCellHeight() / 4;
685                   width = 5 * this.csm.getCellWidth() / 4;
686                   height = this.csm.getCellHeight() / 4;
687                }
688                else if (here.getY() > there.getY())
689                {
690                   //here is south of there
691                   x = (there.getX() - 1) * this.csm.getCellWidth() + this.csm.getCellWidth() / 4;
692                   y = (there.getY() - 1) * this.csm.getCellHeight() + this.csm.getCellHeight() / 4;
693                   width = this.csm.getCellWidth() / 4;
694                   height = this.csm.getCellHeight();
695                }
696                else
697                {

835       private void drawArrow(Graphics2D g, Direction local, int x, int y)
836       {
837          //Draws an arrow in the direction of "local" centered on the point (x,y)
838          if (local.equals(Direction.North))
839          {
840             int[] ys =
841             {
842                y + this.csm.getCellHeight() * 3 / 8, y, y, y - this.csm.getCellHeight() * 3 / 8, y, y
843             };
844             int[] xs =
845             {
846                x, x - this.csm.getCellWidth() / 8, x - this.csm.getCellWidth() / 4, x,
847                x + this.csm.getCellWidth() / 4, x + this.csm.getCellWidth() / 8
848             };
849             g.drawPolygon(xs, ys, 6);
850          }
851          if (local.equals(Direction.South))
852          {
853             int[] ys =
854             {
855                y - this.csm.getCellHeight() * 3 / 8, y, y, y + this.csm.getCellHeight() * 3 / 8, y, y
856             };
857             int[] xs =
858             {
859                x, x - this.csm.getCellWidth() / 8, x - this.csm.getCellWidth() / 4, x,
860                x + this.csm.getCellWidth() / 4, x + this.csm.getCellWidth() / 8
861             };
862             g.drawPolygon(xs, ys, 6);
863          }
864          if (local.equals(Direction.West))
865          {
866             int[] xs =
867             {
868                x + this.csm.getCellWidth() * 3 / 8, x, x, x - this.csm.getCellWidth() * 3 / 8, x, x
869             };
870             int[] ys =
871             {
872                y, y - this.csm.getCellHeight() / 8, y - this.csm.getCellHeight() / 4, y,
873                y + this.csm.getCellHeight() / 4, y + this.csm.getCellHeight() / 8
874             };
875             g.drawPolygon(xs, ys, 6);
876          }
877          if (local.equals(Direction.East))
878          {
879             int[] xs =
880             {
881                x - this.csm.getCellWidth() * 3 / 8, x, x, x + this.csm.getCellWidth() * 3 / 8, x, x
882             };
883             int[] ys =
884             {
885                y, y - this.csm.getCellHeight() / 8, y - this.csm.getCellHeight() / 4, y,
886                y + this.csm.getCellHeight() / 4, y + this.csm.getCellHeight() / 8
887             };
888             g.drawPolygon(xs, ys, 6);
889          }
890       }
