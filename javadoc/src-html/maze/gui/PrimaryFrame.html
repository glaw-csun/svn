239       public MazeInfo saveMaze(MazeInfo mi)
240       {
241          DefaultComboBoxModel cbm = mMazeInfoModel.getMazeInfoComboBoxModel();
242          TreeSet<String> paths = new TreeSet<String>();
243          for (int i = 0; i < cbm.getSize(); i++)
244          {
245             String path = ((MazeInfo) cbm.getElementAt(i)).getPath();
246             if (!path.equals(""))
247                paths.add(path.toLowerCase());
248          }
249          if (mi.isDirty())
250          {
251             MazeInfo toSave = mi;
252             if (!mi.isMutable())
253             {
254                toSave = mi.getMutableClone();
255                Box box = new Box(BoxLayout.Y_AXIS);
256                JLabel label = new JLabel("What would you like ot call the" + " maze?");
257                JTextField field = new JTextField();
258                box.add(label);
259                box.add(field);
260                String newName = null;
261                while (true)
262                {
263                   int result = JOptionPane.showConfirmDialog(PrimaryFrame.this,
264                                                              box,
265                                                              "Maze Name",
266                                                              JOptionPane.OK_CANCEL_OPTION,
267                                                              JOptionPane.QUESTION_MESSAGE);
268                   if (result == JOptionPane.OK_OPTION)
269                   {
270                      if (field.getText().length() > 0)
271                      {
272                         newName = field.getText();
273                         break;
274                      }
275                   }
276                   else
277                      break;
278                }
279                if (newName == null)
280                   return null;
281                mi.clearDirty();
282                toSave.setName(newName);
283                toSave.setPath("");
284                mMazeInfoModel.addMaze(toSave);
285                saveMaze(toSave);
286                return toSave;
287             }
288             else if (mi.getPath() == null || mi.getPath().equals(""))
289             {
290                JFileChooser chooser = new JFileChooser(".");
291                while (true)
292                {
293                   if (chooser.showSaveDialog(this) == JFileChooser.APPROVE_OPTION)
294                   {
295                      File file = chooser.getSelectedFile();
296                      try
297                      {
298                         if (paths.contains(file.getCanonicalPath().toLowerCase()))
299                         {
300                            JOptionPane.showMessageDialog(this,
301                                                          "That file is being used by another maze",
302                                                          "Error",
303                                                          JOptionPane.ERROR_MESSAGE);
304                         }
305                         else
306                         {
307                            mi.setPath(file.getCanonicalPath());
308                            break;
309                         }
310                      }
311                      catch (IOException ex)
312                      {
313                         JOptionPane.showMessageDialog(this,
314                                                       "Invalid Save File",
315                                                       "Error",
316                                                       JOptionPane.ERROR_MESSAGE);
317                      }
318                   } // if (chooser.showSaveDialog(this) ==
319                   //  JFileChooser.APPROVE_OPTION)
320                   else
321                      break;
322                } // while (true)
323             } // if (mi.getPath().equals(""))
324             mi.store();
325          } // if (mi.isDirty())
326          return mi;
327       }

333       @Override
334       public void windowClosing(WindowEvent e)
335       {
336          DefaultComboBoxModel cbm = mMazeInfoModel.getMazeInfoComboBoxModel();
337          for (int i = 0; i < cbm.getSize(); i++)
338          {
339             MazeInfo mi = (MazeInfo) cbm.getElementAt(i);
340             if (mi.isDirty())
341             {
342                int result;
343                result = JOptionPane.showConfirmDialog(this,
344                                                       "Would you like to save \"" +
345                                                             mi.getName() +
346                                                             "\"",
347                                                       "Save Maze?",
348                                                       JOptionPane.YES_NO_OPTION,
349                                                       JOptionPane.QUESTION_MESSAGE);
350                if (result == JOptionPane.YES_NO_OPTION)
351                   saveMaze(mi);
352             }
353          } // for (int i = 0; i < cbm.getSize(); i++)
354       }
