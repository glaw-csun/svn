290       /**
291        * Bring up the open file dialog and based on the file type selected hand it
292        * off to the appropriate panel. The 2 groups of files are AI scripts and
293        * maze files. AI scripts are python .py files and maze maps can be .maz or
294        * .mz2.
295        */
296       private void doOpenFile()
297       {
298          final MenuControlled[] panels =
299          {
300             codeEditorPanel, mazeEditor
301          };
302          final JFileChooser fc = new JFileChooser();
303          fc.setAcceptAllFileFilterUsed(false);
304    
305          // Create a file filter for all our file types.
306          FileFilter allFiles = new FileFilter()
307          {
308             @Override
309             public String getDescription()
310             {
311                String msg = "";
312                for (MenuControlled mc : panels)
313                {
314                   msg += mc.getFileTypeDescription() + ", ";
315                }
316                return msg.substring(0, msg.length() - 2);
317             }
318    
319             @Override
320             public boolean accept(File f)
321             {
322                if (f.isDirectory())
323                   return true;
324                for (MenuControlled mc : panels)
325                {
326                   if (mc.isMyFileType(f))
327                      return true;
328                }
329                return false;
330             }
331          };
332          fc.addChoosableFileFilter(allFiles);
333    
334          // Add filter for each type.
335          for (final MenuControlled mc : panels)
336          {
337             fc.addChoosableFileFilter(new FileFilter()
338             {
339                @Override
340                public boolean accept(File f)
341                {
342                   if (f.isDirectory())
343                      return true;
344                   else
345                      return mc.isMyFileType(f);
346                }
347    
348                @Override
349                public String getDescription()
350                {
351                   return mc.getFileTypeDescription();
352                }
353             });
354          }
355    
356          fc.setFileFilter(allFiles); // Set initially selected.
357    
358          if (fc.showOpenDialog(PrimaryFrame.this) == JFileChooser.APPROVE_OPTION)
359          {
360             File file = fc.getSelectedFile();
361             try
362             {
363                if (file.exists() && file.canRead())
364                {
365                   // Try to open the file so it throws an exception if the file can not be read.
366                   FileInputStream fileStream = null;
367                   try
368                   {
369                      fileStream = new FileInputStream(file);
370                   }
371                   finally
372                   {
373                      if (fileStream != null)
374                         fileStream.close();
375                   }
376                   for (final MenuControlled mc : panels)
377                   {
378                      if (mc.isMyFileType(file))
379                      {
380                         mainTabs.setSelectedComponent((Component) mc);
381                         mc.open(file);
382                         return;
383                      }
384                   }
385                   // If we make it this far then nobody recognized the file type.
386                   throw new RuntimeException("The type of the selected file was not recognized.");
387                }
388                else
389                {
390                   throw new SecurityException("File may not exist.");
391                }
392             }
393             catch (Exception ex)
394             {
395                ex.printStackTrace();
396                String msg = "There was an error opening the file.\n" + ex.getLocalizedMessage();
397                JOptionPane.showMessageDialog(PrimaryFrame.this,
398                                              msg,
399                                              "File Open Error",
400                                              JOptionPane.ERROR_MESSAGE);
401             }
402          }
403       }

415       /**
416        * Exits the application.
417        */
418       public void exit()
419       {
420          if (this.canExitingApplication())
421          {
422             this.setVisible(false);
423             System.exit(0);
424          }
425       }
426    
427       /**
428        * Called when the application is closed. Perform application exit
429        * cleanup.This gives us a chance to cleanup and prompt to save things.
430        * @return true if it is OK for the application to exit. false if the
431        *         application should about the exit.
432        */
433       private boolean canExitingApplication()
434       {
435          return this.mazeEditor.canExit() && this.codeEditorPanel.canExit();
436       }
